% constraints/rules
% 1. A block cannot be on itself.
impossible on(B)=B.

% 2. A block cannot be on two different locations at the same time.
impossible on(B)=L1 & on(B)=L2 & L1\=L2.

% 3. Two different blocks cannot be on the same block at the same time.
impossible on(B1)=B & on(B2)=B & B1\=B2.

% 4. If a block is holding in the hand, then it is not on the table and it is not on any block.
impossible holding(B) & on(B)=L.

% 5. If the hand is empty, then the hand is not holding any block.
impossible handEmpty & holding(B).

% 6. If the hand is holding some block, then the hand is not empty.
impossible holding(B) & handEmpty.

% 7. The hand cannot be holding two different blocks at the same time.
impossible holding(B1) & holding(B2) & B1\=B2.

% 8. A block is clear exactly when there is no block on top of it. (If something is on B, then B is not clear.)
impossible on(B1)=B2 & clear(B2).

% 9. A block is clear exactly when there is no block on top of it. (If nothing is on B, then B is clear.)
clear(B) if ~ (on(B1)=B).

% 10. If a block is on another block, then the two blocks are different.
impossible on(B1)=B2 & B1=B2.

% 11. If a block is on another block, then the block it is on is not clear.
impossible on(B1)=B2 & clear(B2).

% 12. Picking up a block causes the hand to be holding that block.
pickupBlock(B) causes holding(B).

% 13. Picking up a block causes the hand to not be empty.
pickupBlock(B) causes ~handEmpty.

% 14. Picking up a block causes the block to not be on the table.
pickupBlock(B) causes on(B)=L if L\=table.

% 15. Picking up a block causes the block to not be clear.
pickupBlock(B) causes ~clear(B).

% 16. Picking up a block is not executable if the hand is not empty.
nonexecutable pickupBlock(B) if ~handEmpty.

% 17. Picking up a block is not executable if the block is not clear.
nonexecutable pickupBlock(B) if ~clear(B).

% 18. Picking up a block is not executable if the block is not on the table.
nonexecutable pickupBlock(B) if on(B)\=table.

% 19. Picking up a block is not executable if the hand is already holding some block.
nonexecutable pickupBlock(B) if holding(B1).

% 20. Putting down a block causes the block to be on the table.
putdownBlock(B) causes on(B)=table.

% 21. Putting down a block causes the block to be clear.
putdownBlock(B) causes clear(B).

% 22. Putting down a block causes the hand to be empty.
putdownBlock(B) causes handEmpty.

% 23. Putting down a block causes the hand to not be holding that block.
putdownBlock(B) causes ~holding(B).

% 24. Putting down a block is not executable if the hand is not holding that block.
nonexecutable putdownBlock(B) if ~holding(B).

% 25. Stacking a block onto another block causes the first block to be on the second block.
stackBlock(B1,B2) causes on(B1)=B2.

% 26. Stacking a block onto another block causes the first block to be clear.
stackBlock(B1,B2) causes clear(B1).

% 27. Stacking a block onto another block causes the second block to not be clear.
stackBlock(B1,B2) causes ~clear(B2).

% 28. Stacking a block onto another block causes the hand to be empty.
stackBlock(B1,B2) causes handEmpty.

% 29. Stacking a block onto another block causes the hand to not be holding the first block.
stackBlock(B1,B2) causes ~holding(B1).

% 30. Stacking a block onto another block is not executable if the hand is not holding the first block.
nonexecutable stackBlock(B1,B2) if ~holding(B1).

% 31. Stacking a block onto another block is not executable if the second block is not clear.
nonexecutable stackBlock(B1,B2) if ~clear(B2).

% 32. Stacking a block onto another block is not executable if the two blocks are the same.
nonexecutable stackBlock(B1,B2) if B1=B2.

% 33. Unstacking a block from on top of another block causes the hand to be holding the first block.
unstackBlock(B1,B2) causes holding(B1).

% 34. Unstacking a block from on top of another block causes the hand to not be empty.
unstackBlock(B1,B2) causes ~handEmpty.

% 35. Unstacking a block from on top of another block causes the second block to be clear.
unstackBlock(B1,B2) causes clear(B2).

% 36. Unstacking a block from on top of another block causes the first block to not be clear.
unstackBlock(B1,B2) causes ~clear(B1).

% 37. Unstacking a block from on top of another block causes the first block to not be on the second block.
unstackBlock(B1,B2) causes on(B1)=L if L\=B2.

% 38. Unstacking a block from on top of another block is not executable if the hand is not empty.
nonexecutable unstackBlock(B1,B2) if ~handEmpty.

% 39. Unstacking a block from on top of another block is not executable if the first block is not clear.
nonexecutable unstackBlock(B1,B2) if ~clear(B1).

% 40. Unstacking a block from on top of another block is not executable if the first block is not on the second block.
nonexecutable unstackBlock(B1,B2) if on(B1)\=B2.

% 41. Unstacking a block from on top of another block is not executable if the two blocks are the same.
nonexecutable unstackBlock(B1,B2) if B1=B2.

% 42. It is not executable to pick up a block and put down a block at the same time.
nonexecutable pickupBlock(B1) & putdownBlock(B2) if true.

% 43. It is not executable to pick up a block and stack a block at the same time.
nonexecutable pickupBlock(B1) & stackBlock(B2,B3) if true.

% 44. It is not executable to pick up a block and unstack a block at the same time.
nonexecutable pickupBlock(B1) & unstackBlock(B2,B3) if true.

% 45. It is not executable to put down a block and stack a block at the same time.
nonexecutable putdownBlock(B1) & stackBlock(B2,B3) if true.

% 46. It is not executable to put down a block and unstack a block at the same time.
nonexecutable putdownBlock(B1) & unstackBlock(B2,B3) if true.

% 47. It is not executable to stack a block and unstack a block at the same time.
nonexecutable stackBlock(B1,B2) & unstackBlock(B3,B4) if true.


% query
:- query
    0: handEmpty
       & on(b3)=table & clear(b3)
       & on(b1)=table & clear(b1)
       & on(b2)=table
       & on(b4)=b2 & clear(b4);
    maxstep: on(b3)=table & clear(b3)
             & on(b2)=table
             & on(b4)=b2
             & on(b1)=b4 & clear(b1).
