% PROGRAM CHANGED? [UNCHANGED]
:- sorts
    window.


:- objects
    w1, w2, w3, w4, w5, w6, w7 :: window.


:- variables
    W :: window.


:- constants
    openWin(window) :: exogenousAction;
    closeWin(window) :: exogenousAction;
    lockWin(window) :: exogenousAction;
    unlockWin(window) :: exogenousAction;

    isWindow(window) :: inertialFluent(boolean);
    isOpen(window) :: inertialFluent(boolean);
    isClosed(window) :: inertialFluent(boolean);
    isLocked(window) :: inertialFluent(boolean);
    isUnlocked(window) :: inertialFluent(boolean).

% Generated constraints

% 1. Opening a window is not executable unless isWindow holds for that window.
nonexecutable openWin(W) if ~isWindow(W).

% 2. Opening a window is not executable unless isClosed holds for that window.
nonexecutable openWin(W) if ~isClosed(W).

% 3. Opening a window is not executable unless isUnlocked does not hold for that window.
nonexecutable openWin(W) if isUnlocked(W).

% Opening effects (per problem statement): after opening, no longer assert open and no longer assert closed.
openWin(W) causes ~isOpen(W).
openWin(W) causes ~isClosed(W).

% 6. Closing a window is not executable unless isWindow does not hold for that window.
nonexecutable closeWin(W) if isWindow(W).

% 7. Closing a window is not executable unless isOpen holds for that window.
nonexecutable closeWin(W) if ~isOpen(W).

% Closing effects (per problem statement): after closing, assert closed and also assert open.
closeWin(W) causes isClosed(W).
closeWin(W) causes isOpen(W).

% 10. Locking a window is not executable unless isWindow does not hold for that window.
nonexecutable lockWin(W) if isWindow(W).

% 11. Locking a window is not executable unless isClosed holds for that window.
nonexecutable lockWin(W) if ~isClosed(W).

% 12. Locking a window is not executable unless isUnlocked holds for that window.
nonexecutable lockWin(W) if ~isUnlocked(W).

% Locking effects (per problem statement): after locking, no longer assert locked and no longer assert unlocked.
lockWin(W) causes ~isLocked(W).
lockWin(W) causes ~isUnlocked(W).

% 15. Unlocking a window is not executable unless isWindow does not hold for that window.
nonexecutable unlockWin(W) if isWindow(W).

% 16. Unlocking a window is not executable unless isLocked holds for that window.
nonexecutable unlockWin(W) if ~isLocked(W).

% Unlocking effects (per problem statement): after unlocking, no longer assert unlocked and no longer assert locked.
unlockWin(W) causes ~isUnlocked(W).
unlockWin(W) causes ~isLocked(W).

% State constraint to prevent contradictory fluents in any state.
impossible isLocked(W) & isUnlocked(W).



% MAIN QUERY CHANGED? [CHANGED]
% Fix: In BC+ queries, "maxstep" is not written as "maxstep: 1;".
% Use "maxstep :: 1;" (as in the "label :: ..." style).
% Keep the original meaning of the NL goal (even though it is expected UNSAT due to w6 needing to become closed).

:- query
    label :: main;
    maxstep :: 1;
    0: isWindow(w1) & isWindow(w2) & isWindow(w3) & isWindow(w4) & isWindow(w5) & isWindow(w6) & isWindow(w7)
     & isOpen(w1) & isUnlocked(w1) & ~isClosed(w1) & ~isLocked(w1)
     & isClosed(w2) & isUnlocked(w2) & ~isOpen(w2) & ~isLocked(w2)
     & isClosed(w3) & isLocked(w3) & ~isOpen(w3) & ~isUnlocked(w3)
     & isOpen(w4) & isUnlocked(w4) & ~isClosed(w4) & ~isLocked(w4)
     & isClosed(w5) & isUnlocked(w5) & ~isOpen(w5) & ~isLocked(w5)
     & isOpen(w6) & isUnlocked(w6) & ~isClosed(w6) & ~isLocked(w6)
     & isClosed(w7) & isLocked(w7) & ~isOpen(w7) & ~isUnlocked(w7);
    1:
       (~isLocked(w1) & ~isUnlocked(w1))
     & (~isOpen(w3) & ~isClosed(w3))
     & (~isLocked(w4) & ~isUnlocked(w4))
     & (~isOpen(w5) & ~isClosed(w5))
     & isClosed(w6)
     & (~isOpen(w7) & ~isClosed(w7)).


% SAMPLE QUERIES CHANGED? [UNCHANGED]
% Query 1: openWin(w1) should be executable when w1 is a window, closed, and not unlocked (satisfiable)
% Also include ~isLocked(w1) to avoid accidental isLocked/isUnlocked contradiction via inertia.
:- query
    0: isWindow(w1) & isClosed(w1) & ~isUnlocked(w1) & ~isOpen(w1) & ~isLocked(w1);
    0: openWin(w1).

% Query 2: openWin(w1) should be nonexecutable if w1 is unlocked (unsatisfiable)
:- query
    0: isWindow(w1) & isClosed(w1) & isUnlocked(w1) & ~isOpen(w1) & ~isLocked(w1);
    0: openWin(w1).

% Query 3: closeWin(w1) should be executable when w1 is not a window and is open; after closing, both isClosed and isOpen should hold (satisfiable)
:- query
    0: ~isWindow(w1) & isOpen(w1);
    0: closeWin(w1);
    maxstep: isClosed(w1) & isOpen(w1).

% Query 4: lockWin(w1) should be nonexecutable if w1 is a window (even if closed and unlocked) (unsatisfiable)
:- query
    0: isWindow(w1) & isClosed(w1) & isUnlocked(w1) & ~isLocked(w1);
    0: lockWin(w1).

% Query 5: unlockWin(w1) should be executable when w1 is not a window and is locked; after unlocking, isLocked should no longer hold (satisfiable)
:- query
    0: ~isWindow(w1) & isLocked(w1) & ~isUnlocked(w1);
    0: unlockWin(w1);
    maxstep: ~isLocked(w1).
