:- sorts
    ant;
    place.


:- objects
    ant1 :: ant;
    p1, p2 :: place.


:- variables
    A :: ant;
    P, P1, P2, P3, P4 :: place.


:- constants
    move(ant, place, place) :: exogenousAction;
    pickUpFood(ant, place) :: exogenousAction;
    dropFoodAtNest(ant, place) :: exogenousAction;
    clearObstacle(ant, place, place) :: exogenousAction;

    loc(ant) :: inertialFluent(place);
    adjacent(place, place) :: inertialFluent(boolean);
    hasFood(place) :: inertialFluent(boolean);
    isNest(place) :: inertialFluent(boolean);
    clear(place) :: inertialFluent(boolean);
    carryingFood(ant) :: inertialFluent(boolean).

% Generated constraints

% 1. An ant cannot be at two different places simultaneously.
impossible loc(A)=P1 & loc(A)=P2 & P1\=P2.

% 2. Moving an ant from a starting place to a destination place causes the ant's location to become the destination place.
move(A,P1,P2) causes loc(A)=P2.

% 3. Moving is not executable if the ant is not at the starting place.
nonexecutable move(A,P1,P2) if loc(A)\=P1.

% 4. Moving is not executable if the starting place is not adjacent to the destination place.
nonexecutable move(A,P1,P2) if ~adjacent(P1,P2).

% 5. Moving is not executable if the destination place IS clear.
% (Natural language: move can occur when destination is not clear.)
nonexecutable move(A,P1,P2) if clear(P2).

% 6. Picking up food at a place causes the ant to be carrying food.
pickUpFood(A,P) causes carryingFood(A).

% 7. Picking up food at a place does not change whether that place has food.
% NOTE: Redundant due to inertia of hasFood, but harmless.
pickUpFood(A,P) causes hasFood(P) if hasFood(P).

% 8. Picking up food is not executable if the ant IS at that place.
% (Natural language: pickUpFood can occur when the ant is not at the place.)
nonexecutable pickUpFood(A,P) if loc(A)=P.

% 9. Picking up food is not executable if that place does not have food.
nonexecutable pickUpFood(A,P) if ~hasFood(P).

% 10. Picking up food is not executable if the ant is already carrying food.
nonexecutable pickUpFood(A,P) if carryingFood(A).

% 11. Dropping food at a place causes that place to have food.
dropFoodAtNest(A,P) causes hasFood(P).

% 12. Dropping food at nest keeps the ant carrying food.
% NOTE: Redundant due to inertia of carryingFood, but harmless.
dropFoodAtNest(A,P) causes carryingFood(A).

% 13. Dropping food is not executable if the ant IS at that place.
% (Natural language: dropFoodAtNest can occur when the ant is not at the place.)
nonexecutable dropFoodAtNest(A,P) if loc(A)=P.

% 14. Dropping food is not executable if that place is not a nest.
nonexecutable dropFoodAtNest(A,P) if ~isNest(P).

% 15. Dropping food is not executable if the ant is not carrying food.
nonexecutable dropFoodAtNest(A,P) if ~carryingFood(A).

% 16. Clearing an obstacle should make the blocked place clear.
clearObstacle(A,P1,P2) causes clear(P2).

% 17. Clearing an obstacle is not executable if the ant IS at the starting place.
% (Natural language: clearObstacle can occur when the ant is not at the starting place.)
nonexecutable clearObstacle(A,P1,P2) if loc(A)=P1.

% 18. Clearing an obstacle is not executable if the starting place is not adjacent to the blocked place.
nonexecutable clearObstacle(A,P1,P2) if ~adjacent(P1,P2).

% 19. Clearing an obstacle is not executable if the blocked place is clear.
nonexecutable clearObstacle(A,P1,P2) if clear(P2).

% 20. It is not executable to perform two different move actions at the same time for the same ant.
nonexecutable move(A,P1,P2) & move(A,P3,P4) if (P1\=P3 | P2\=P4).

% 21. It is not executable to both move and pick up food at the same time for the same ant.
nonexecutable move(A,P1,P2) & pickUpFood(A,P).

% 22. It is not executable to both move and drop food at a nest at the same time for the same ant.
nonexecutable move(A,P1,P2) & dropFoodAtNest(A,P).

% 23. It is not executable to both move and clear an obstacle at the same time for the same ant.
nonexecutable move(A,P1,P2) & clearObstacle(A,P3,P4).

% 24. It is not executable to both pick up food and drop food at a nest at the same time for the same ant.
nonexecutable pickUpFood(A,P1) & dropFoodAtNest(A,P2).

% 25. It is not executable to both pick up food and clear an obstacle at the same time for the same ant.
nonexecutable pickUpFood(A,P1) & clearObstacle(A,P2,P3).

% 26. It is not executable to both drop food at a nest and clear an obstacle at the same time for the same ant.
nonexecutable dropFoodAtNest(A,P1) & clearObstacle(A,P2,P3).

% [CHANGED] Fix: adjacency should be rigid, not inertial.
% Otherwise, the solver can freely change adjacent/2 over time (as seen in outputs where
% adjacent(p1,p1), adjacent(p2,p2), and isNest(p2) appear without being stated).
% This change makes the transition system match commonsense: adjacency is a static map.

:- constants
    adjacent(place, place) :: rigid(boolean).

% [CHANGED] Fix: isNest should be rigid as well (nest-ness should not change by inertia).

:- constants
    isNest(place) :: rigid(boolean).


noconcurrency.

:- query
    0: loc(ant1)=p1
       & adjacent(p1,p2) & adjacent(p2,p1)
       & clear(p1) & ~clear(p2)
       & isNest(p1)
       & ~hasFood(p1)
       & hasFood(p2)
       & ~carryingFood(ant1);
    maxstep: hasFood(p1).
