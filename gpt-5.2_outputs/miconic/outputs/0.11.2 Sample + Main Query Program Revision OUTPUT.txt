% PROGRAM CHANGED? [CHANGED]
% BC+ signature
:- sorts
    floor;
    passenger.

:- objects
    f0, f1 :: floor;
    p0 :: passenger.

:- variables
    F, F1, F2, F3, F4 :: floor;
    P, P1, P2 :: passenger.

:- constants
    board(passenger,floor) :: exogenousAction;
    depart(passenger,floor) :: exogenousAction;
    moveUp(floor,floor) :: exogenousAction;
    moveDown(floor,floor) :: exogenousAction;

    elevAt(floor) :: inertialFluent(boolean);
    origin(passenger) :: inertialFluent(floor);
    dest(passenger) :: inertialFluent(floor);
    above(floor,floor) :: inertialFluent(boolean);
    boarded(passenger) :: inertialFluent(boolean);
    served(passenger) :: inertialFluent(boolean).

% Generated constraints

% Elevator cannot be at two different floors at the same time.
impossible elevAt(F1) & elevAt(F2) & F1\=F2.

% Elevator must be at at least one floor at all times (otherwise it could be at no floor).
impossible ~elevAt(f0) & ~elevAt(f1).

% Boarding effects.
board(P,F) causes boarded(P).

% Boarding executability conditions.
nonexecutable board(P,F) if ~elevAt(F).
nonexecutable board(P,F) if origin(P)\=F.
nonexecutable board(P,F) if boarded(P).
nonexecutable board(P,F) if served(P).

% Departing effects.
depart(P,F) causes ~boarded(P).
depart(P,F) causes served(P).

% Departing executability conditions.
nonexecutable depart(P,F) if ~elevAt(F).
nonexecutable depart(P,F) if dest(P)\=F.
nonexecutable depart(P,F) if ~boarded(P).
nonexecutable depart(P,F) if served(P).

% Moving up effects.
moveUp(F1,F2) causes elevAt(F2).
moveUp(F1,F2) causes ~elevAt(F1).

% Moving up executability conditions.
nonexecutable moveUp(F1,F2) if ~elevAt(F1).
nonexecutable moveUp(F1,F2) if ~above(F1,F2).
nonexecutable moveUp(F1,F2) if F1=F2.

% Moving down effects.
moveDown(F1,F2) causes elevAt(F2).
moveDown(F1,F2) causes ~elevAt(F1).

% Moving down executability conditions.
nonexecutable moveDown(F1,F2) if ~elevAt(F1).
nonexecutable moveDown(F1,F2) if ~above(F2,F1).
nonexecutable moveDown(F1,F2) if F1=F2.

% Concurrency restrictions.

% Not executable to move up and move down at the same time.
nonexecutable moveUp(F1,F2) & moveDown(F3,F4) if true.

% Not executable to perform two different moveUp actions at the same time.
% NOTE: parenthesize disjunction in the body.
nonexecutable moveUp(F1,F2) & moveUp(F3,F4) if (F1\=F3 | F2\=F4).

% Not executable to perform two different moveDown actions at the same time.
% NOTE: parenthesize disjunction in the body.
nonexecutable moveDown(F1,F2) & moveDown(F3,F4) if (F1\=F3 | F2\=F4).

% Not executable to both move (up or down) and board at the same time.
nonexecutable moveUp(F1,F2) & board(P,F) if true.
nonexecutable moveDown(F1,F2) & board(P,F) if true.

% Not executable to both move (up or down) and depart at the same time.
nonexecutable moveUp(F1,F2) & depart(P,F) if true.
nonexecutable moveDown(F1,F2) & depart(P,F) if true.

% Not executable for the same passenger to both board and depart at the same time.
nonexecutable board(P,F1) & depart(P,F2) if true.

% Fix: prevent the solver from freely making "above/2" true.
% Make above/2 rigid by forcing it to persist across transitions.
above(F1,F2) after above(F1,F2).
~above(F1,F2) after ~above(F1,F2).

% Additional fix: prevent the solver from freely making "origin/1" and "dest/1" change.
% These are intended to be fixed passenger data, not something actions can modify.
origin(P)=F after origin(P)=F.
dest(P)=F after dest(P)=F.

% Additional fix: prevent the solver from freely making "served/1" true without departing.
% (Inertial fluents can otherwise be chosen arbitrarily in the initial state unless constrained.)
default ~served(P).
default ~boarded(P).

% MAIN QUERY CHANGED? [UNCHANGED]
:- query
    0: above(f0,f1) & origin(p0)=f1 & dest(p0)=f0 & elevAt(f0) & ~elevAt(f1) & ~boarded(p0) & ~served(p0);
    maxstep: served(p0).

% SAMPLE QUERIES CHANGED? [UNCHANGED]
% Query 1: Board at the correct origin floor when the elevator is there, then depart at the correct destination after moving there (satisfiable)
:- query
    label :: q1_board_move_depart;
    0: above(f0,f1) & origin(p0)=f1 & dest(p0)=f0 &
       elevAt(f1) & ~elevAt(f0) & ~boarded(p0) & ~served(p0);
    0: board(p0,f1);
    1: moveDown(f1,f0);
    2: depart(p0,f0);
    maxstep: served(p0) & ~boarded(p0).

% Query 2: Try to board when the elevator is not at that floor (unsatisfiable)
:- query
    label :: q2_board_wrong_floor;
    0: above(f0,f1) & origin(p0)=f1 & dest(p0)=f0 &
       elevAt(f0) & ~elevAt(f1) & ~boarded(p0) & ~served(p0);
    0: board(p0,f1).

% Query 3: Try to board at a floor that is not the passenger's origin (unsatisfiable)
:- query
    label :: q3_board_not_origin;
    0: above(f0,f1) & origin(p0)=f1 & dest(p0)=f0 &
       elevAt(f0) & ~elevAt(f1) & ~boarded(p0) & ~served(p0);
    0: board(p0,f0).

% Query 4: Try to depart without having boarded first (unsatisfiable)
:- query
    label :: q4_depart_without_boarding;
    0: above(f0,f1) & origin(p0)=f1 & dest(p0)=f0 &
       elevAt(f0) & ~elevAt(f1) & ~boarded(p0) & ~served(p0);
    0: depart(p0,f0).

% Query 5: Try to move up from f1 to f0 even though above(f1,f0) is false (given only above(f0,f1)) (unsatisfiable)
:- query
    label :: q5_moveup_not_above;
    0: above(f0,f1) & ~above(f1,f0) & origin(p0)=f1 & dest(p0)=f0 &
       elevAt(f1) & ~elevAt(f0) & ~boarded(p0) & ~served(p0);
    0: moveUp(f1,f0).
