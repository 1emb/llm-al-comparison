% constraints/rules
% 1. The elevator is always at exactly one floor (it cannot be at two different floors at the same time).
impossible elevAt(F1) & elevAt(F2) & F1\=F2.

% 2. Boarding a passenger at a floor causes that passenger to be boarded.
board(P,F) causes boarded(P).

% 3. Boarding a passenger at a floor is not executable if the elevator is not at that floor.
nonexecutable board(P,F) if ~elevAt(F).

% 4. Boarding a passenger at a floor is not executable if that floor is not the passenger's origin.
nonexecutable board(P,F) if origin(P)\=F.

% 5. Boarding a passenger at a floor is not executable if the passenger is already boarded.
nonexecutable board(P,F) if boarded(P).

% 6. Boarding a passenger at a floor is not executable if the passenger is already served.
nonexecutable board(P,F) if served(P).

% 7. Departing a passenger at a floor causes that passenger to no longer be boarded.
depart(P,F) causes ~boarded(P).

% 8. Departing a passenger at a floor causes that passenger to be served.
depart(P,F) causes served(P).

% 9. Departing a passenger at a floor is not executable if the elevator is not at that floor.
nonexecutable depart(P,F) if ~elevAt(F).

% 10. Departing a passenger at a floor is not executable if that floor is not the passenger's destination.
nonexecutable depart(P,F) if dest(P)\=F.

% 11. Departing a passenger at a floor is not executable if the passenger is not boarded.
nonexecutable depart(P,F) if ~boarded(P).

% 12. Departing a passenger at a floor is not executable if the passenger is already served.
nonexecutable depart(P,F) if served(P).

% 13. Moving up from a starting floor to a destination floor causes the elevator to be at the destination floor.
moveUp(F1,F2) causes elevAt(F2).

% 14. Moving up from a starting floor to a destination floor causes the elevator to no longer be at the starting floor.
moveUp(F1,F2) causes ~elevAt(F1).

% 15. Moving up from a starting floor to a destination floor is not executable if the elevator is not at the starting floor.
nonexecutable moveUp(F1,F2) if ~elevAt(F1).

% 16. Moving up from a starting floor to a destination floor is not executable if the starting floor is not above the destination floor.
nonexecutable moveUp(F1,F2) if ~above(F1,F2).

% 17. Moving up from a starting floor to a destination floor is not executable if the starting floor and destination floor are the same floor.
nonexecutable moveUp(F1,F2) if F1=F2.

% 18. Moving down from a starting floor to a destination floor causes the elevator to be at the destination floor.
moveDown(F1,F2) causes elevAt(F2).

% 19. Moving down from a starting floor to a destination floor causes the elevator to no longer be at the starting floor.
moveDown(F1,F2) causes ~elevAt(F1).

% 20. Moving down from a starting floor to a destination floor is not executable if the elevator is not at the starting floor.
nonexecutable moveDown(F1,F2) if ~elevAt(F1).

% 21. Moving down from a starting floor to a destination floor is not executable if the destination floor is not above the starting floor.
nonexecutable moveDown(F1,F2) if ~above(F2,F1).

% 22. Moving down from a starting floor to a destination floor is not executable if the starting floor and destination floor are the same floor.
nonexecutable moveDown(F1,F2) if F1=F2.

% 23. It is not executable to move up and move down at the same time.
nonexecutable moveUp(F1,F2) & moveDown(F3,F4) if true.

% 24. It is not executable to perform two different move up actions at the same time.
nonexecutable moveUp(F1,F2) & moveUp(F3,F4) if F1\=F3 | F2\=F4.

% 25. It is not executable to perform two different move down actions at the same time.
nonexecutable moveDown(F1,F2) & moveDown(F3,F4) if F1\=F3 | F2\=F4.

% 26. It is not executable to both move (up or down) and board a passenger at the same time.
nonexecutable moveUp(F1,F2) & board(P,F) if true.
nonexecutable moveDown(F1,F2) & board(P,F) if true.

% 27. It is not executable to both move (up or down) and depart a passenger at the same time.
nonexecutable moveUp(F1,F2) & depart(P,F) if true.
nonexecutable moveDown(F1,F2) & depart(P,F) if true.

% 28. It is not executable for the same passenger to both board and depart at the same time.
nonexecutable board(P,F1) & depart(P,F2) if true.

% query
:- query
    0: above(f0,f1) & origin(p0)=f1 & dest(p0)=f0 & elevAt(f0) & ~boarded(p0) & ~served(p0);
    maxstep: served(p0).
