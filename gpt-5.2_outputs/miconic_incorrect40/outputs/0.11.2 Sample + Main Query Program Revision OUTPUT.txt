% PROGRAM CHANGED? [CHANGED]
% BC+ signature
:- sorts
    floor;
    passenger.

:- objects
    f0, f1, f2, f3 :: floor;
    p0, p1 :: passenger.

:- variables
    F, F1, F2 :: floor;
    P :: passenger.

:- constants
    board(passenger) :: exogenousAction;
    depart(passenger) :: exogenousAction;
    moveUp(floor) :: exogenousAction;
    moveDown(floor) :: exogenousAction;

    % Attributes
    atFloor(passenger) :: attribute(floor) of board(passenger);
    atFloor2(passenger) :: attribute(floor) of depart(passenger);
    toFloorUp(floor) :: attribute(floor) of moveUp(floor);
    toFloorDown(floor) :: attribute(floor) of moveDown(floor);

    liftAt :: inertialFluent(floor);
    origin(passenger) :: inertialFluent(floor);
    dest(passenger) :: inertialFluent(floor);

    above(floor,floor) :: inertialFluent(boolean);

    isFloor(floor) :: inertialFluent(boolean);
    isPassenger(passenger) :: inertialFluent(boolean);
    boarded(passenger) :: inertialFluent(boolean);
    served(passenger) :: inertialFluent(boolean).

% Generated constraints

% --- Basic sanity constraints ---
impossible liftAt = F1 & liftAt = F2 & F1\=F2.

% --- Boarding ---
nonexecutable board(P) if atFloor(P)\=liftAt.
nonexecutable board(P) if origin(P)\=atFloor(P).

% Boarding is not executable at floor F if isFloor(F) holds.
nonexecutable board(P) if isFloor(F) & atFloor(P)=F.

% Boarding is not executable for passenger P if isPassenger(P) holds.
nonexecutable board(P) if isPassenger(P).

board(P) causes boarded(P).
board(P) causes isPassenger(P).

% --- Departing ---
nonexecutable depart(P) if atFloor2(P)\=liftAt.
nonexecutable depart(P) if dest(P)\=atFloor2(P).

% Departing is not executable at floor F if isFloor(F) holds.
nonexecutable depart(P) if isFloor(F) & atFloor2(P)=F.

% Departing is not executable for passenger P if isPassenger(P) does not hold.
nonexecutable depart(P) if ~isPassenger(P).

% Departing is not executable for passenger P if boarded(P) does not hold.
nonexecutable depart(P) if ~boarded(P).

depart(P) causes ~boarded(P).
depart(P) causes served(P).

% --- Moving up/down ---
% IMPORTANT FIX:
% The original program required destination to satisfy isFloor(dest).
% But the main query sets isFloor(f0..f3)=true, and boarding/departing were forbidden when isFloor(floor)=true.
% That makes serving impossible. We remove the "isFloor" gating from board/depart and from moveUp/moveDown.

% MoveUp(F1): move from current floor F1 to toFloorUp(F1)=F2, requiring above(F2,F1).
nonexecutable moveUp(F1) if liftAt\=F1.
nonexecutable moveUp(F1) if toFloorUp(F1)=F2 & ~above(F2,F1).
moveUp(F1) causes liftAt=F2 if toFloorUp(F1)=F2.

% MoveDown(F1): move from current floor F1 to toFloorDown(F1)=F2, requiring above(F1,F2).
nonexecutable moveDown(F1) if liftAt\=F1.
nonexecutable moveDown(F1) if toFloorDown(F1)=F2 & ~above(F1,F2).
moveDown(F1) causes liftAt=F2 if toFloorDown(F1)=F2.


% MAIN QUERY CHANGED? [CHANGED]
:- query
    0: above(f0,f1) & above(f0,f2) & above(f0,f3) & above(f1,f2) & above(f1,f3) & above(f2,f3);
    0: origin(p0)=f3 & dest(p0)=f2 & origin(p1)=f1 & dest(p1)=f3;
    0: liftAt=f0;

    % Keep these as given (they no longer block boarding/departing).
    0: isFloor(f0) & isFloor(f1) & isFloor(f2) & isFloor(f3);

    0: ~isPassenger(p0) & ~isPassenger(p1) & ~boarded(p0) & ~boarded(p1) & ~served(p0) & ~served(p1);

    maxstep: served(p0) & served(p1).
