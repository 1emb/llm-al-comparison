% BC+ signature
:- sorts
    ant;
    place.

:- objects
    ant1 :: ant;
    p1, p2 :: place.

:- variables
    A, A1, A2, A3, A4 :: ant;
    P, P1, P2, P3, P4 :: place;
    Bool, Bool1, Bool2, Bool3, Bool4 :: boolean.

:- constants
    % Actions
    move(ant, place, place) :: exogenousAction;
    pickupFood(ant, place) :: exogenousAction;
    dropFood(ant, place) :: exogenousAction;
    clearObstacle(ant, place, place) :: exogenousAction;

    % Fluents
    loc(ant) :: inertialFluent(place);
    adjacent(place, place) :: inertialFluent(boolean);
    clear(place) :: inertialFluent(boolean);
    isNest(place) :: inertialFluent(boolean);
    hasFood(place) :: inertialFluent(boolean);
    carryingFood(ant) :: inertialFluent(boolean).

% Generated constraints/rules

% Moving an ant from a source place to a target place causes the ant's location to become the target place.
move(A,P1,P2) causes loc(A)=P2.

% Moving an ant from a source place to a target place is not executable if the ant is not at the source place.
nonexecutable move(A,P1,P2) if loc(A)\=P1.

% Moving an ant from a source place to a target place is not executable if the source place and target place are not adjacent.
nonexecutable move(A,P1,P2) if ~adjacent(P1,P2).

% Moving an ant from a source place to a target place is not executable if the target place is not clear.
nonexecutable move(A,P1,P2) if ~clear(P2).

% Picking up food at a place causes the ant to be carrying food.
pickupFood(A,P) causes carryingFood(A).

% Picking up food at a place causes that place to no longer have food.
pickupFood(A,P) causes ~hasFood(P).

% Picking up food at a place is not executable if the ant is not at that place.
nonexecutable pickupFood(A,P) if loc(A)\=P.

% Picking up food at a place is not executable if the ant is already carrying food.
nonexecutable pickupFood(A,P) if carryingFood(A).

% Picking up food at a place is not executable if that place does not have food.
nonexecutable pickupFood(A,P) if ~hasFood(P).

% Dropping food at a place causes the ant to not be carrying food.
dropFood(A,P) causes ~carryingFood(A).

% Dropping food at a place is not executable if the ant is not at that place.
nonexecutable dropFood(A,P) if loc(A)\=P.

% Dropping food at a place is not executable if the ant is not carrying food.
nonexecutable dropFood(A,P) if ~carryingFood(A).

% Dropping food at a place is not executable if that place is not a nest.
nonexecutable dropFood(A,P) if ~isNest(P).

% Clearing an obstacle at a target place from a source place causes the target place to be clear.
clearObstacle(A,P1,P2) causes clear(P2).

% Clearing an obstacle at a target place from a source place is not executable if the ant is not at the source place.
nonexecutable clearObstacle(A,P1,P2) if loc(A)\=P1.

% Clearing an obstacle at a target place from a source place is not executable if the source place and target place are not adjacent.
nonexecutable clearObstacle(A,P1,P2) if ~adjacent(P1,P2).

% Clearing an obstacle at a target place from a source place is not executable if the target place is already clear.
nonexecutable clearObstacle(A,P1,P2) if clear(P2).

% It is not executable for the same ant to both move and pick up food in the same step.
nonexecutable move(A,P1,P2) & pickupFood(A,P) if true.

% It is not executable for the same ant to both move and drop food in the same step.
nonexecutable move(A,P1,P2) & dropFood(A,P) if true.

% It is not executable for the same ant to both move and clear an obstacle in the same step.
nonexecutable move(A,P1,P2) & clearObstacle(A,P3,P4) if true.

% It is not executable for the same ant to both pick up food and drop food in the same step.
nonexecutable pickupFood(A,P1) & dropFood(A,P2) if true.

% It is not executable for the same ant to both pick up food and clear an obstacle in the same step.
nonexecutable pickupFood(A,P1) & clearObstacle(A,P2,P3) if true.

% It is not executable for the same ant to both drop food and clear an obstacle in the same step.
nonexecutable dropFood(A,P1) & clearObstacle(A,P2,P3) if true.

% NOTE: The Cplus2ASP outputs show isNest(p2) and adjacent(p1,p1)/adjacent(p2,p2) appearing even when not stated.
% To avoid unintended models, make adjacency and nesthood explicit constraints rather than freely chosen inertial fluents.

% Places are not adjacent to themselves.
impossible adjacent(P,P).

% Only p1 is a nest (for this 2-place instance).
isNest(p1).
impossible isNest(p2).

       & clear(p1) & clear(p2)
       & isNest(p1)
       & hasFood(p2)
       & ~carryingFood(ant1);
    maxstep: ~carryingFood(ant1) & ~hasFood(p2) & loc(ant1)=p1.
