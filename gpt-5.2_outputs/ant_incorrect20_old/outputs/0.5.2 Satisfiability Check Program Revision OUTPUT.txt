% BC+ signature
:- sorts
    ant;
    place.

:- objects
    ant1 :: ant;
    p1, p2 :: place.

:- variables
    A :: ant;
    P, P1, P2, P3, P4 :: place.

:- constants
    move(ant, place, place) :: exogenousAction;
    pickUpFood(ant, place) :: exogenousAction;
    dropFoodAtNest(ant, place) :: exogenousAction;
    clearObstacle(ant, place, place) :: exogenousAction;

    % Use a functional fluent for location to avoid the syntax error from at(ant,place)
    % and to naturally enforce "an ant is at exactly one place".
    loc(ant) :: inertialFluent(place);

    adjacent(place, place) :: inertialFluent(boolean);
    clear(place) :: inertialFluent(boolean);
    hasFood(place) :: inertialFluent(boolean);
    isNest(place) :: inertialFluent(boolean);
    carryingFood(ant) :: inertialFluent(boolean).

% Generated constraints

% Moving an ant from a starting place to a destination place causes the ant to be at the destination place.
move(A,P1,P2) causes loc(A)=P2.

% The move action is not executable unless the ant is currently at the starting place.
nonexecutable move(A,P1,P2) if loc(A)\=P1.

% The move action is not executable if the starting place is adjacent to the destination place.
nonexecutable move(A,P1,P2) if adjacent(P1,P2).

% The move action is not executable if the destination place is not clear.
nonexecutable move(A,P1,P2) if ~clear(P2).

% The move action is not executable if the starting place and the destination place are the same place.
nonexecutable move(A,P1,P2) if P1=P2.

% Picking up food at a place causes the ant to be carrying food.
pickUpFood(A,P) causes carryingFood(A).

% Picking up food at a place causes the place to no longer have food.
pickUpFood(A,P) causes ~hasFood(P).

% The pick-up-food action is not executable unless the ant is at that place.
nonexecutable pickUpFood(A,P) if loc(A)\=P.

% The pick-up-food action is not executable if that place does not have food.
nonexecutable pickUpFood(A,P) if ~hasFood(P).

% The pick-up-food action is not executable if the ant is already carrying food.
nonexecutable pickUpFood(A,P) if carryingFood(A).

% Dropping food at a place causes the ant to no longer be carrying food.
dropFoodAtNest(A,P) causes ~carryingFood(A).

% Dropping food at a place causes the place to have food.
dropFoodAtNest(A,P) causes hasFood(P).

% The drop-food-at-nest action is not executable unless the ant is at that place.
nonexecutable dropFoodAtNest(A,P) if loc(A)\=P.

% The drop-food-at-nest action is not executable if that place is not a nest.
nonexecutable dropFoodAtNest(A,P) if ~isNest(P).

% The drop-food-at-nest action is not executable if the ant is not carrying food.
nonexecutable dropFoodAtNest(A,P) if ~carryingFood(A).

% Clearing an obstacle at a blocked place while at a current place causes the blocked place to become clear.
clearObstacle(A,P1,P2) causes clear(P2).

% The clear-obstacle action is not executable unless the ant is at the current place.
nonexecutable clearObstacle(A,P1,P2) if loc(A)\=P1.

% The clear-obstacle action is not executable if the current place is adjacent to the blocked place.
nonexecutable clearObstacle(A,P1,P2) if adjacent(P1,P2).

% The clear-obstacle action is not executable if the blocked place is already clear.
nonexecutable clearObstacle(A,P1,P2) if clear(P2).

% The clear-obstacle action is not executable if the current place and the blocked place are the same place.
nonexecutable clearObstacle(A,P1,P2) if P1=P2.

% Concurrency constraints (same intent as original)

% It is not executable for the same ant to perform two different move actions at the same time.
nonexecutable move(A,P1,P2) & move(A,P3,P4) if (P1\=P3 | P2\=P4).

% It is not executable for the same ant to move and pick up food at the same time.
nonexecutable move(A,P1,P2) & pickUpFood(A,P3) if true.

% It is not executable for the same ant to move and drop food at a nest at the same time.
nonexecutable move(A,P1,P2) & dropFoodAtNest(A,P3) if true.

% It is not executable for the same ant to move and clear an obstacle at the same time.
nonexecutable move(A,P1,P2) & clearObstacle(A,P3,P4) if true.

% It is not executable for the same ant to pick up food and drop food at a nest at the same time.
nonexecutable pickUpFood(A,P1) & dropFoodAtNest(A,P2) if true.

% It is not executable for the same ant to pick up food and clear an obstacle at the same time.
nonexecutable pickUpFood(A,P1) & clearObstacle(A,P2,P3) if true.

% It is not executable for the same ant to drop food at a nest and clear an obstacle at the same time.
nonexecutable dropFoodAtNest(A,P1) & clearObstacle(A,P2,P3) if true.

% Main query
% Original natural language query: Ant ant1 is at p1. p1 is adjacent to p2 and p2 is adjacent to p1.
% p1 is clear and p2 is clear. p1 is the nest. Food is present at p2.
% Find a plan to bring the food to the nest.
:- query
    0: loc(ant1)=p1
       & adjacent(p1,p2) & adjacent(p2,p1)
       & clear(p1) & clear(p2)
       & isNest(p1) & ~isNest(p2)
       & hasFood(p2) & ~hasFood(p1)
       & ~carryingFood(ant1);
    maxstep: hasFood(p1).
