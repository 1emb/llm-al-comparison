:- sorts
    loc >> disk;
    loc >> peg.


:- objects
    a, b, c :: peg;
    d1, d2, d3 :: disk.


:- variables
    D, D1, D2, D3, D4 :: disk;
    P, P1, P2, P3, P4 :: peg;
    L, L1, L2, L3, L4 :: loc;
    N, N1, N2, N3, N4 :: integer;
    Bool, Bool1, Bool2, Bool3, Bool4 :: boolean.


:- constants
    on(disk) :: inertialFluent(loc);
    weight(disk) :: inertialFluent(integer);
    move(disk, loc) :: exogenousAction.


% constraints/rules

% 1. Each disk is on exactly one location (either a peg or another disk) at any time.
impossible on(D)=L1 & on(D)=L2 & L1\=L2.

% 2. It is impossible for two different disks to be on the same disk at the same time.
impossible on(D1)=D & on(D2)=D & D1\=D2.

% 3. Moving a disk to a location causes that disk to be on that destination location.
move(D,L) causes on(D)=L.

% 4. A move of a disk to a location is not executable if the disk is already on that location.
nonexecutable move(D,L) if on(D)=L.

% 5. A disk cannot be moved if there is another disk on top of it (i.e., if some other disk is on it).
nonexecutable move(D,L) if on(D1)=D.

% 6. A disk cannot be moved onto another disk if it is heavier than that destination disk.
nonexecutable move(D1,D2) if weight(D1)=N1 & weight(D2)=N2 & N1 > N2.

% 7. If a disk is moved onto another disk, then after the move the destination disk has that disk on top of it.
% (Captured by rule 3 together with the meaning of on/1: after move(D1,D2), on(D1)=D2.)

% 8. It is not executable to perform two different move actions involving the same disk at the same time.
nonexecutable move(D,L1) & move(D,L2) if L1\=L2.

% 9. It is not executable to move two different disks onto the same destination disk at the same time.
nonexecutable move(D1,D) & move(D2,D) if D1\=D2.



% query
:- query
    0: on(d1)=a & on(d2)=d1 & on(d3)=d2 & weight(d1)=3 & weight(d2)=2 & weight(d3)=1;
    maxstep: on(d1)=c & on(d2)=d1 & on(d3)=d2.
