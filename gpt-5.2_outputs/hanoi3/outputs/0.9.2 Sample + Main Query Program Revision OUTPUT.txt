% PROGRAM CHANGED? [CHANGED]
% BC+ signature
:- sorts
    loc;
    disk;
    peg;
    integer.

:- objects
    a, b, c :: peg;
    d1, d2, d3 :: disk;
    0..10 :: integer.

:- variables
    D, D1, D2, D3 :: disk;
    P :: peg;
    L :: loc;
    N :: integer.

:- constants
    loc(disk) :: inertialFluent(loc);
    weight(disk) :: inertialFluent(integer);
    move(disk, loc) :: exogenousAction.

% Generated constraints / commonsense constraints

% --- sort membership / domain-closure for loc ---
% loc can only be a peg or a disk
impossible loc(D)=L & L\=a & L\=b & L\=c & L\=d1 & L\=d2 & L\=d3.

% 1. A disk cannot be on itself.
impossible loc(D)=D.

% 2. At most one disk can be directly on any disk.
impossible loc(D1)=D & loc(D2)=D & D1\=D2.

% (Optional but standard for Towers-of-Hanoi style domains)
% At most one disk can be directly on any peg.
impossible loc(D1)=P & loc(D2)=P & D1\=D2.

% 3. Moving a disk to a location causes that disk’s location to become the destination.
move(D,L) causes loc(D)=L.

% 4. A disk cannot be moved if there is another disk on top of it.
nonexecutable move(D,L) if loc(D1)=D.

% 5. Moving a disk onto a disk is not executable if the moving disk is heavier than the destination disk.
nonexecutable move(D1,D2) if weight(D1) > weight(D2).

% 6. Moving a disk onto a disk is not executable if there is already a disk on top of the destination disk.
nonexecutable move(D1,D2) if loc(D3)=D2.

% 7. Moving a disk to a location is not executable if the disk is already at that location.
nonexecutable move(D,L) if loc(D)=L.

% NOTE: The Cplus2ASP output “Error: The specified query was not found.”
% is not a logical satisfiable/unsatisfiable result; it indicates the tool
% did not find the query label you asked it to run (or was invoked without
% selecting a query). The BC+ program itself is fine; the issue is with how
% queries were run/selected, not with these laws.

% MAIN QUERY CHANGED? [UNCHANGED]
% Original natural language query: Disk d1 is on peg a. Disk d2 is on disk d1. Disk d3 is on disk d2. In order of heaviest to lightest, the disks are d1, d2, and d3. Find a plan to stack all disks on peg c.
:- query
    0: loc(d1)=a & loc(d2)=d1 & loc(d3)=d2 & weight(d1)=3 & weight(d2)=2 & weight(d3)=1;
    maxstep: loc(d1)=c & loc(d2)=d1 & loc(d3)=d2.

% SAMPLE QUERIES CHANGED? [UNCHANGED]
% Query 1: Move the top disk d3 from d2 onto peg b (should be allowed) (satisfiable)
:- query
    label :: q1;
    0: loc(d1)=a & loc(d2)=d1 & loc(d3)=d2 & weight(d1)=3 & weight(d2)=2 & weight(d3)=1;
    0: move(d3,b).

% Query 2: Try to move disk d2 while d3 is on top of it (should be blocked) (unsatisfiable)
:- query
    label :: q2;
    0: loc(d1)=a & loc(d2)=d1 & loc(d3)=d2 & weight(d1)=3 & weight(d2)=2 & weight(d3)=1;
    0: move(d2,b).

% Query 3: Try to move heavier disk d1 onto lighter disk d3 (should violate weight constraint) (unsatisfiable)
:- query
    label :: q3;
    0: loc(d1)=a & loc(d2)=b & loc(d3)=c & weight(d1)=3 & weight(d2)=2 & weight(d3)=1;
    0: move(d1,d3).

% Query 4: Try to move d1 onto d2 when d2 already has a disk on it (d3 on d2), so destination has a disk on top (unsatisfiable)
:- query
    label :: q4;
    0: loc(d1)=a & loc(d2)=b & loc(d3)=d2 & weight(d1)=3 & weight(d2)=2 & weight(d3)=1;
    0: move(d1,d2).

% Query 5: Try to move a disk to the location it is already at (should be disallowed) (unsatisfiable)
:- query
    label :: q5;
    0: loc(d1)=a & loc(d2)=d1 & loc(d3)=d2 & weight(d1)=3 & weight(d2)=2 & weight(d3)=1;
    0: move(d1,a).
