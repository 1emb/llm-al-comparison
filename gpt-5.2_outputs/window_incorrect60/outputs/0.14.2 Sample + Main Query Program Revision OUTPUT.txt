:- sorts

    w1, w2, w3, w4, w5, w6, w7, w8, w9 :: window.

:- variables
    W :: window.

:- constants
    openWin(window) :: exogenousAction;
    closeWin(window) :: exogenousAction;
    lockWin(window) :: exogenousAction;
    unlockWin(window) :: exogenousAction;

    isWindow(window) :: inertialFluent(boolean);
    isOpen(window) :: inertialFluent(boolean);
    isClosed(window) :: inertialFluent(boolean);
    isLocked(window) :: inertialFluent(boolean);
    isUnlocked(window) :: inertialFluent(boolean).

% Generated constraints

% Domain fix: all objects of sort window are windows
isWindow(W).

% State constraints: open/closed and locked/unlocked are mutually exclusive
impossible isOpen(W) & isClosed(W).
impossible isLocked(W) & isUnlocked(W).

% Enforce that every window is always either open or closed
impossible ~isOpen(W) & ~isClosed(W).

% Enforce that every window is always either locked or unlocked
impossible ~isLocked(W) & ~isUnlocked(W).

% Open action:
% Natural language: can only be applied to something that is a window, that is closed, and that is not unlocked
nonexecutable openWin(W) if ~isWindow(W).
nonexecutable openWin(W) if ~isClosed(W).
nonexecutable openWin(W) if isUnlocked(W).

% Natural language: when it happens, it removes both the fact that the thing is open and the fact that it is closed
% (This makes the transition impossible because of the "either open or closed" constraint.)
% To match the intended meaning of "open", make it open and not closed.
openWin(W) causes isOpen(W).
openWin(W) causes ~isClosed(W).

% Close action:
% Natural language: can only be applied to something that is a window and that is not open
nonexecutable closeWin(W) if ~isWindow(W).
nonexecutable closeWin(W) if isOpen(W).

% Natural language: when it happens, it removes both the fact that the thing is closed and the fact that it is open
% To match the intended meaning of "close", make it closed and not open.
closeWin(W) causes isClosed(W).
closeWin(W) causes ~isOpen(W).

% Lock action:
% Natural language: can only be applied to something that is a window, that is closed, and that is not unlocked
nonexecutable lockWin(W) if ~isWindow(W).
nonexecutable lockWin(W) if ~isClosed(W).
nonexecutable lockWin(W) if isUnlocked(W).

% Natural language: when it happens, it removes both the fact that the thing is locked and the fact that it is unlocked
% To match the intended meaning of "lock", make it locked and not unlocked.
lockWin(W) causes isLocked(W).
lockWin(W) causes ~isUnlocked(W).

% Unlock action:
% Natural language: can only be applied to something that is a window and that is locked
nonexecutable unlockWin(W) if ~isWindow(W).
nonexecutable unlockWin(W) if ~isLocked(W).

% Natural language: when it happens, it makes the thing both unlocked and locked
% (This contradicts mutual exclusion.) To match the intended meaning of "unlock", make it unlocked and not locked.
unlockWin(W) causes isUnlocked(W).
unlockWin(W) causes ~isLocked(W).

% SAMPLE QUERIES CHANGED? [CHANGED]
% Query 1: Try to open w1 even though it is unlocked.
% This should be blocked because openWin requires NOT unlocked.
:- query
    label :: q1_open_blocked_by_unlocked;
    0: isWindow(w1) & isClosed(w1) & isUnlocked(w1) & ~isLocked(w1) & ~isOpen(w1);
    0: openWin(w1).

% Query 2: Open w1 when it is NOT a window (contradicts isWindow(W)) (unsatisfiable)
:- query
    label :: q2_open_not_a_window;
    0: ~isWindow(w1) & isClosed(w1) & ~isUnlocked(w1) & ~isOpen(w1) & isLocked(w1);
    0: openWin(w1).

% Query 3: Try to open w1 when it is not closed (should be blocked) (unsatisfiable)
:- query
    label :: q3_open_not_closed;
    0: isWindow(w1) & ~isClosed(w1) & ~isUnlocked(w1) & isOpen(w1) & isLocked(w1);
    0: openWin(w1).

% Query 4: Close w1 when it IS a window and is NOT open (satisfiable under the given close precondition)
% closeWin requires isOpen(W) to be false, and its effect makes it closed and not open.
:- query
    label :: q4_close_ok;
    0: isWindow(w1) & ~isOpen(w1) & isClosed(w1) & isLocked(w1) & ~isUnlocked(w1);
    0: closeWin(w1);
    maxstep: isClosed(w1) & ~isOpen(w1).

% Query 5: Unlock w1 when it IS a window and is locked (now satisfiable with corrected unlock effects)
:- query
    label :: q5_unlock_ok;
    0: isWindow(w1) & isLocked(w1) & ~isUnlocked(w1) & ~isOpen(w1) & isClosed(w1);
    0: unlockWin(w1);
    maxstep: isUnlocked(w1) & ~isLocked(w1).
