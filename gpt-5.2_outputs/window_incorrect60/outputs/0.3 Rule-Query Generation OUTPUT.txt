% constraints/rules
% 1. Opening a window is not executable if the thing is a window.
nonexecutable openWin(W) if isWindow(W).

% 2. Opening a window is not executable if the thing is not closed.
nonexecutable openWin(W) if ~isClosed(W).

% 3. Opening a window is not executable if the thing is unlocked.
nonexecutable openWin(W) if isUnlocked(W).

% 4. Opening a window causes the thing to not be open.
openWin(W) causes ~isOpen(W).

% 5. Opening a window causes the thing to not be closed.
openWin(W) causes ~isClosed(W).

% 6. Closing a window is not executable if the thing is not a window.
nonexecutable closeWin(W) if ~isWindow(W).

% 7. Closing a window is not executable if the thing is open.
nonexecutable closeWin(W) if isOpen(W).

% 8. Closing a window causes the thing to not be closed.
closeWin(W) causes ~isClosed(W).

% 9. Closing a window causes the thing to not be open.
closeWin(W) causes ~isOpen(W).

% 10. Locking a window is not executable if the thing is a window.
nonexecutable lockWin(W) if isWindow(W).

% 11. Locking a window is not executable if the thing is not closed.
nonexecutable lockWin(W) if ~isClosed(W).

% 12. Locking a window is not executable if the thing is unlocked.
nonexecutable lockWin(W) if isUnlocked(W).

% 13. Locking a window causes the thing to not be locked.
lockWin(W) causes ~isLocked(W).

% 14. Locking a window causes the thing to not be unlocked.
lockWin(W) causes ~isUnlocked(W).

% 15. Unlocking a window is not executable if the thing is a window.
nonexecutable unlockWin(W) if isWindow(W).

% 16. Unlocking a window is not executable if the thing is not locked.
nonexecutable unlockWin(W) if ~isLocked(W).

% 17. Unlocking a window causes the thing to be unlocked.
unlockWin(W) causes isUnlocked(W).

% 18. Unlocking a window causes the thing to be locked.
unlockWin(W) causes isLocked(W).

% query
:- query
    0: isOpen(w1) & isUnlocked(w1) & ~isClosed(w1) & ~isLocked(w1);
    0: isClosed(w2) & isUnlocked(w2) & ~isOpen(w2) & ~isLocked(w2);
    0: isClosed(w3) & isLocked(w3) & ~isOpen(w3) & ~isUnlocked(w3);
    0: isOpen(w4) & isUnlocked(w4) & ~isClosed(w4) & ~isLocked(w4);
    0: isClosed(w5) & isUnlocked(w5) & ~isOpen(w5) & ~isLocked(w5);
    0: isOpen(w6) & isUnlocked(w6) & ~isClosed(w6) & ~isLocked(w6);
    0: isClosed(w7) & isLocked(w7) & ~isOpen(w7) & ~isUnlocked(w7);
    0: isClosed(w8) & isUnlocked(w8) & ~isOpen(w8) & ~isLocked(w8);
    0: isOpen(w9) & isUnlocked(w9) & ~isClosed(w9) & ~isLocked(w9);
    maxstep: isLocked(w1) & isOpen(w3) & isLocked(w4) & isOpen(w5) & isClosed(w6) & isOpen(w7) & isLocked(w9).
