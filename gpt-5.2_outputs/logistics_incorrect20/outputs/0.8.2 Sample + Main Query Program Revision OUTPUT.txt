% Generated constraints

% A package cannot be inside two different vehicles at the same time.
impossible in(P,V1) & in(P,V2) & V1\=V2.

% If a package is inside a vehicle, it is at the same place as that vehicle.
loc(P)=Pl if in(P,V) & loc(V)=Pl.

% Load into truck: makes package inside truck and (redundantly) at that location.
loadTruck(P,T,L) causes in(P,T).
loadTruck(P,T,L) causes loc(P)=L.
nonexecutable loadTruck(P,T,L) if loc(T)\=L.
nonexecutable loadTruck(P,T,L) if loc(P)\=L.
nonexecutable loadTruck(P,T,L) if in(P,V).

% Load into airplane: makes package inside airplane and at that airport.
loadPlane(P,A,Ap) causes in(P,A).
loadPlane(P,A,Ap) causes loc(P)=Ap.
nonexecutable loadPlane(P,A,Ap) if loc(A)\=Ap.
nonexecutable loadPlane(P,A,Ap) if loc(P)\=Ap.
nonexecutable loadPlane(P,A,Ap) if in(P,V).

% Unload from truck: removes inside relation and sets package location.
unloadTruck(P,T,L) causes ~in(P,T).
unloadTruck(P,T,L) causes loc(P)=L.
nonexecutable unloadTruck(P,T,L) if loc(T)\=L.
nonexecutable unloadTruck(P,T,L) if ~in(P,T).

% Unload from airplane: removes inside relation and sets package location.
unloadPlane(P,A,Ap) causes ~in(P,A).
unloadPlane(P,A,Ap) causes loc(P)=Ap.
nonexecutable unloadPlane(P,A,Ap) if loc(A)\=Ap.
nonexecutable unloadPlane(P,A,Ap) if ~in(P,A).

% Drive truck within a city: changes truck location.
drive(T,L1,L2,C) causes loc(T)=L2.
nonexecutable drive(T,L1,L2,C) if ~inCity(L1,C).
nonexecutable drive(T,L1,L2,C) if ~inCity(L2,C).
nonexecutable drive(T,L1,L2,C) if loc(T)\=L1.

% Fly airplane between airports: changes airplane location.
fly(A,Ap1,Ap2) causes loc(A)=Ap2.
nonexecutable fly(A,Ap1,Ap2) if loc(A)\=Ap1.

% Concurrency constraints (must include "if true" to avoid the parser treating them as "noconcurrency")
nonexecutable loadTruck(P,T,L) & unloadTruck(P,T,L) if true.
nonexecutable loadPlane(P,A,Ap) & unloadPlane(P,A,Ap) if true.

nonexecutable drive(T,L1,L2,C) & drive(T,L1,L3,C) if L2\=L3.
nonexecutable fly(A,Ap1,Ap2) & fly(A,Ap1,Ap3) if Ap2\=Ap3.
