:- sorts
    vehicle >> truck;
    vehicle >> airplane;
    thing >> package;
    thing >> vehicle;
    place >> location;
    location >> airport;
    city.



:- objects
    c0, c1 :: city;
    l00, l10 :: airport;
    t0, t1 :: truck;
    a0 :: airplane;
    p0 :: package.



:- variables
    P, P1, P2 :: package;
    T, T1 :: truck;
    A, A1 :: airplane;
    V, V1, V2 :: vehicle;
    Pl, Pl1, Pl2 :: place;
    L, L1, L2, L3 :: location;
    Ap, Ap1, Ap2, Ap3 :: airport;
    C, C1 :: city.



:- constants
    loadTruck(package, truck, location) :: exogenousAction;
    loadPlane(package, airplane, airport) :: exogenousAction;
    unloadTruck(package, truck, location) :: exogenousAction;
    unloadPlane(package, airplane, airport) :: exogenousAction;
    drive(truck, location, location, city) :: exogenousAction;
    fly(airplane, airport, airport) :: exogenousAction;

    loc(thing) :: inertialFluent(place);
    in(package, vehicle) :: inertialFluent(boolean);
    inCity(place, city) :: inertialFluent(boolean).

% Generated constraints

% A package cannot be inside two different vehicles at the same time.
impossible in(P,V1) & in(P,V2) & V1\=V2.

% If a package is inside a vehicle, it is at the same place as that vehicle.
loc(P)=Pl if in(P,V) & loc(V)=Pl.

% A thing cannot be at two different places at the same time.
impossible loc(P)=Pl1 & loc(P)=Pl2 & Pl1\=Pl2.
impossible loc(T)=Pl1 & loc(T)=Pl2 & Pl1\=Pl2.
impossible loc(A)=Pl1 & loc(A)=Pl2 & Pl1\=Pl2.

% Load into truck: makes package inside truck.
loadTruck(P,T,L) causes in(P,T).
nonexecutable loadTruck(P,T,L) if loc(T)\=L.
nonexecutable loadTruck(P,T,L) if loc(P)\=L.
nonexecutable loadTruck(P,T,L) if in(P,V).

% Load into airplane: makes package inside airplane.
loadPlane(P,A,Ap) causes in(P,A).
nonexecutable loadPlane(P,A,Ap) if loc(A)\=Ap.
nonexecutable loadPlane(P,A,Ap) if loc(P)\=Ap.
nonexecutable loadPlane(P,A,Ap) if in(P,V).

% Unload from truck: removes inside relation.
unloadTruck(P,T,L) causes ~in(P,T).
nonexecutable unloadTruck(P,T,L) if loc(T)\=L.
nonexecutable unloadTruck(P,T,L) if ~in(P,T).

% Unload from airplane: removes inside relation.
unloadPlane(P,A,Ap) causes ~in(P,A).
nonexecutable unloadPlane(P,A,Ap) if loc(A)\=Ap.
nonexecutable unloadPlane(P,A,Ap) if ~in(P,A).

% Drive truck within a city: changes truck location.
drive(T,L1,L2,C) causes loc(T)=L2.
nonexecutable drive(T,L1,L2,C) if ~inCity(L1,C).
nonexecutable drive(T,L1,L2,C) if ~inCity(L2,C).
nonexecutable drive(T,L1,L2,C) if loc(T)\=L1.

% Fly airplane between airports: changes airplane location.
fly(A,Ap1,Ap2) causes loc(A)=Ap2.
nonexecutable fly(A,Ap1,Ap2) if loc(A)\=Ap1.

% Concurrency constraints (must include "if true" to avoid the parser treating them as "noconcurrency")
nonexecutable loadTruck(P,T,L) & unloadTruck(P,T,L) if true.
nonexecutable loadPlane(P,A,Ap) & unloadPlane(P,A,Ap) if true.

nonexecutable drive(T,L1,L2,C) & drive(T,L1,L3,C) if L2\=L3.
nonexecutable fly(A,Ap1,Ap2) & fly(A,Ap1,Ap3) if Ap2\=Ap3.

% Make inCity false by default (so it is not arbitrarily chosen at step 0).
default ~inCity(Pl,C).

% Make in false by default (so it is not arbitrarily chosen at step 0).
default ~in(P,V).

% Prevent "teleporting" by driving/flying: destination must differ from origin.
nonexecutable drive(T,L1,L2,C) if L1=L2.
nonexecutable fly(A,Ap1,Ap2) if Ap1=Ap2.

% [CHANGED] Critical fix: driving must not be allowed between different cities.
% Otherwise, the solver can "choose" inCity(l10,c0) at step 0 (since it was not ruled out),
% and then drive t1 from l10 (in c1) to l00 (in c0) using city c0, as seen in the main output.
impossible inCity(Pl,C) & inCity(Pl,C1) & C\=C1.



noconcurrency.

:- query
    0: inCity(l00,c0) & inCity(l10,c1) &
       loc(a0)=l00 & loc(t0)=l00 & loc(t1)=l10 & loc(p0)=l10;
    maxstep: loc(p0)=l00.
