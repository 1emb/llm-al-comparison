% BC+ signature
:- sorts
    ant;
    place.

:- objects
    ant1 :: ant;
    p1, p2 :: place.

:- variables
    A :: ant;
    P, P1, P2, P3, P4 :: place.

:- constants
    % actions
    move(ant, place, place) :: exogenousAction;
    pickUpFood(ant, place) :: exogenousAction;
    dropFoodAtNest(ant, place) :: exogenousAction;
    clearObstacle(ant, place, place) :: exogenousAction;

    % fluents
    loc(ant) :: inertialFluent(place);
    hasFood(ant) :: inertialFluent(boolean);
    hasFoodAt(place) :: inertialFluent(boolean);
    isNest(place) :: inertialFluent(boolean);
    isClear(place) :: inertialFluent(boolean);

    % rigid relation (NOTE: rigidFluent is not a valid declaration type in this setting)
    adjacent(place, place) :: inertialFluent(boolean).

% Generated constraints

% 1. An ant cannot be at two different places at the same time.
impossible loc(A)=P1 & loc(A)=P2 & P1\=P2.

% 2. Moving an ant from a starting place to a destination place causes the ant to be at the destination place.
move(A,P1,P2) causes loc(A)=P2.

% 3. The move action is not executable if the ant is not at the stated starting place.
nonexecutable move(A,P1,P2) if loc(A)\=P1.

% 4. The move action is not executable if the stated starting place is not adjacent to the stated destination.
nonexecutable move(A,P1,P2) if ~adjacent(P1,P2).

% 5. The move action is not executable if the stated destination is clear.
nonexecutable move(A,P1,P2) if isClear(P2).

% 6. Picking up food by an ant at a place causes the ant to have food.
pickUpFood(A,P) causes hasFood(A).

% 7. Picking up food by an ant at a place causes the place to no longer have food.
pickUpFood(A,P) causes ~hasFoodAt(P).

% 8. The pick-up-food action is not executable if the ant is not at the specified place.
nonexecutable pickUpFood(A,P) if loc(A)\=P.

% 9. The pick-up-food action is not executable if the specified place does not have food.
nonexecutable pickUpFood(A,P) if ~hasFoodAt(P).

% 10. The pick-up-food action is not executable if the ant already has food.
nonexecutable pickUpFood(A,P) if hasFood(A).

% 11. Dropping food at a nest by an ant at a place causes the ant to no longer have food.
dropFoodAtNest(A,P) causes ~hasFood(A).

% 12. Dropping food at a nest by an ant at a place causes the place to have food.
dropFoodAtNest(A,P) causes hasFoodAt(P).

% 13. The drop-food-at-nest action is not executable if the ant is not at the specified place.
nonexecutable dropFoodAtNest(A,P) if loc(A)\=P.

% 14. The drop-food-at-nest action is not executable if the specified place is not a nest.
nonexecutable dropFoodAtNest(A,P) if ~isNest(P).

% 15. The drop-food-at-nest action is not executable if the ant does not have food.
nonexecutable dropFoodAtNest(A,P) if ~hasFood(A).

% 16. Clearing an obstacle at a blocked place from a starting place causes the blocked place to no longer be clear.
clearObstacle(A,P1,P2) causes ~isClear(P2).

% 17. The clear-obstacle action is not executable if the ant is not at the stated starting place.
nonexecutable clearObstacle(A,P1,P2) if loc(A)\=P1.

% 18. The clear-obstacle action is not executable if the stated starting place is not adjacent to the blocked place.
nonexecutable clearObstacle(A,P1,P2) if ~adjacent(P1,P2).

% 19. The clear-obstacle action is not executable if the blocked place is not clear.
nonexecutable clearObstacle(A,P1,P2) if ~isClear(P2).

% 20. It is not executable for an ant to perform two different move actions at the same time.
nonexecutable move(A,P1,P2) & move(A,P3,P4) if (P1\=P3 | P2\=P4).

% 21. It is not executable for an ant to move and pick up food at the same time.
nonexecutable move(A,P1,P2) & pickUpFood(A,P3).

% 22. It is not executable for an ant to move and drop food at a nest at the same time.
nonexecutable move(A,P1,P2) & dropFoodAtNest(A,P3).

% 23. It is not executable for an ant to move and clear an obstacle at the same time.
nonexecutable move(A,P1,P2) & clearObstacle(A,P3,P4).

% 24. It is not executable for an ant to pick up food and drop food at a nest at the same time.
nonexecutable pickUpFood(A,P1) & dropFoodAtNest(A,P2).

% 25. It is not executable for an ant to pick up food and clear an obstacle at the same time.
nonexecutable pickUpFood(A,P1) & clearObstacle(A,P2,P3).

% 26. It is not executable for an ant to drop food at a nest and clear an obstacle at the same time.
nonexecutable dropFoodAtNest(A,P1) & clearObstacle(A,P2,P3).

% Main query
% Original natural language query: Ant ant1 is at p1. p1 is adjacent to p2 and p2 is adjacent to p1.
% p1 is clear and p2 is clear. p1 is the nest. Food is present at p2.
% Find a plan to bring the food to the nest.
:- query
    0: loc(ant1)=p1 & adjacent(p1,p2) & adjacent(p2,p1) & isClear(p1) & isClear(p2) & isNest(p1) & hasFoodAt(p2) & ~hasFood(ant1);
    maxstep: hasFoodAt(p1).
