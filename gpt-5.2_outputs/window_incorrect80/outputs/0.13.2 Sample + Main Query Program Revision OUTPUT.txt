% PROGRAM CHANGED? [CHANGED]
% BC+ signature
:- sorts
    window.

:- objects
    w1, w2, w3, w4, w5, w6, w7, w8, w9, w10, w11 :: window.

:- variables
    W :: window.

:- constants
    openWin(window) :: exogenousAction;
    closeWin(window) :: exogenousAction;
    lockWin(window) :: exogenousAction;
    unlockWin(window) :: exogenousAction;

    isWindow(window) :: inertialFluent(boolean);
    isOpen(window) :: inertialFluent(boolean);
    isClosed(window) :: inertialFluent(boolean);
    isLocked(window) :: inertialFluent(boolean);
    isUnlocked(window) :: inertialFluent(boolean).

% Generated constraints

% Make open/closed and locked/unlocked mutually exclusive in every state.
impossible isOpen(W) & isClosed(W).
impossible isLocked(W) & isUnlocked(W).

% Ensure each window is always in exactly one of each pair.
impossible ~isOpen(W) & ~isClosed(W).
impossible ~isLocked(W) & ~isUnlocked(W).

% Actions apply only when the item is explicitly NOT a window.
% Since all objects are of sort window, we enforce ~isWindow(W) in every state.
impossible isWindow(W).

% Therefore, actions are nonexecutable if isWindow(W) holds (redundant but kept for clarity).
nonexecutable openWin(W) if isWindow(W).
nonexecutable closeWin(W) if isWindow(W).
nonexecutable lockWin(W) if isWindow(W).
nonexecutable unlockWin(W) if isWindow(W).

% Opening can occur only when: not a window, closed, and not unlocked.
nonexecutable openWin(W) if ~isClosed(W).
nonexecutable openWin(W) if isUnlocked(W).

% After opening: made closed and made not open. (As stated, even though it is counterintuitive.)
openWin(W) causes isClosed(W).
openWin(W) causes ~isOpen(W).

% Closing can occur only when: not a window and not open.
nonexecutable closeWin(W) if isOpen(W).

% After closing: made open and made not closed. (As stated, even though it is counterintuitive.)
closeWin(W) causes isOpen(W).
closeWin(W) causes ~isClosed(W).

% Locking can occur only when: not a window, closed, and not unlocked.
nonexecutable lockWin(W) if ~isClosed(W).
nonexecutable lockWin(W) if isUnlocked(W).

% After locking: made unlocked and made not locked. (As stated, even though it is counterintuitive.)
lockWin(W) causes isUnlocked(W).
lockWin(W) causes ~isLocked(W).

% Unlocking can occur only when: not a window and not locked.
nonexecutable unlockWin(W) if isLocked(W).

% After unlocking: made locked and made not unlocked. (As stated, even though it is counterintuitive.)
unlockWin(W) causes isLocked(W).
unlockWin(W) causes ~isUnlocked(W).

% NOTE ABOUT UNSAT:
% With the above effects, it is impossible to ever make isLocked(W) true by executing lockWin(W),
% because lockWin(W) causes ~isLocked(W). Similarly, openWin(W) causes ~isOpen(W), etc.
% So many goals that require isLocked(...) or isOpen(...) will be unreachable from some initial states.
% This matches the given (counterintuitive) natural language, and explains UNSAT for the main query.


% MAIN QUERY CHANGED? [UNCHANGED]
:- query
    0: isOpen(w1) & ~isClosed(w1) & isUnlocked(w1) & ~isLocked(w1) &
       isClosed(w2) & ~isOpen(w2) & isUnlocked(w2) & ~isLocked(w2) &
       isClosed(w3) & ~isOpen(w3) & isLocked(w3) & ~isUnlocked(w3) &
       isOpen(w4) & ~isClosed(w4) & isUnlocked(w4) & ~isLocked(w4) &
       isClosed(w5) & ~isOpen(w5) & isUnlocked(w5) & ~isLocked(w5) &
       isOpen(w6) & ~isClosed(w6) & isUnlocked(w6) & ~isLocked(w6) &
       isClosed(w7) & ~isOpen(w7) & isLocked(w7) & ~isUnlocked(w7) &
       isOpen(w8) & ~isClosed(w8) & isUnlocked(w8) & ~isLocked(w8) &
       isClosed(w9) & ~isOpen(w9) & isUnlocked(w9) & ~isLocked(w9) &
       isClosed(w10) & ~isOpen(w10) & isLocked(w10) & ~isUnlocked(w10) &
       isOpen(w11) & ~isClosed(w11) & isUnlocked(w11) & ~isLocked(w11) &
       ~isWindow(w1) & ~isWindow(w2) & ~isWindow(w3) & ~isWindow(w4) & ~isWindow(w5) &
       ~isWindow(w6) & ~isWindow(w7) & ~isWindow(w8) & ~isWindow(w9) & ~isWindow(w10) &
       ~isWindow(w11);
    maxstep: isLocked(w1) &
             isOpen(w3) &
             isLocked(w4) &
             isOpen(w5) &
             isClosed(w6) &
             isOpen(w7) &
             isLocked(w8) &
             isOpen(w9) &
             isOpen(w10) &
             isClosed(w11).

% Sample 1 (satisfiable): a state can exist with w1 open and unlocked (and not a window).
:- query
    label :: sample_sat_state;
    0: isOpen(w1) & ~isClosed(w1) & isUnlocked(w1) & ~isLocked(w1) & ~isWindow(w1);
    maxstep: isOpen(w1) & isUnlocked(w1) & ~isWindow(w1).

% Sample 2 (unsatisfiable): cannot be both open and closed.
:- query
    label :: sample_unsat_open_closed;
    0: isOpen(w1) & isClosed(w1) & ~isWindow(w1);
    maxstep: isOpen(w1).

% Sample 3 (unsatisfiable): cannot ever have isWindow(w1) due to impossible isWindow(W).
:- query
    label :: sample_unsat_isWindow;
    0: isWindow(w1);
    maxstep: isWindow(w1).

% Sample 4 (unsatisfiable): trying to achieve isLocked(w1) from an initially unlocked state is impossible

:- query
    label :: sample_unsat_make_locked_from_unlocked;
    0: isUnlocked(w1) & ~isLocked(w1) & ~isWindow(w1) & isClosed(w1) & ~isOpen(w1);
    maxstep: isLocked(w1).
