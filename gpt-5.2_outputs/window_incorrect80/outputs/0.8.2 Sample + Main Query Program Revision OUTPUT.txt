% Generated constraints
% NOTE: The natural language says actions apply only when the item is explicitly NOT a window.
% Since all objects are of sort window, we model this by requiring ~isWindow(W) as a precondition.

% Make open/closed and locked/unlocked mutually exclusive in every state.
impossible isOpen(W) & isClosed(W).
impossible isLocked(W) & isUnlocked(W).

% (Optional but matches intended meaning of "may be described as open or closed" etc.)
% Ensure each window is always in exactly one of each pair.
impossible ~isOpen(W) & ~isClosed(W).
impossible ~isLocked(W) & ~isUnlocked(W).

% 1. Opening is nonexecutable unless the item is not a window, it is closed, and it is not unlocked.
nonexecutable openWin(W) if isWindow(W).
nonexecutable openWin(W) if ~isClosed(W).
nonexecutable openWin(W) if isUnlocked(W).

% 2-3. After opening: made closed and made not open.
openWin(W) causes isClosed(W).
openWin(W) causes ~isOpen(W).

% 4. Closing is nonexecutable unless the item is not a window and it is not open.
nonexecutable closeWin(W) if isWindow(W).
nonexecutable closeWin(W) if isOpen(W).

% 5-6. After closing: made open and made not closed.
closeWin(W) causes isOpen(W).
closeWin(W) causes ~isClosed(W).

% 7. Locking is nonexecutable unless the item is not a window, it is closed, and it is not unlocked.
nonexecutable lockWin(W) if isWindow(W).
nonexecutable lockWin(W) if ~isClosed(W).
nonexecutable lockWin(W) if isUnlocked(W).

% 8-9. After locking: made unlocked and made not locked.
lockWin(W) causes isUnlocked(W).
lockWin(W) causes ~isLocked(W).

% 10. Unlocking is nonexecutable unless the item is not a window and it is not locked.
nonexecutable unlockWin(W) if isWindow(W).
nonexecutable unlockWin(W) if isLocked(W).

% 11-12. After unlocking: made locked and made not unlocked.
unlockWin(W) causes isLocked(W).
unlockWin(W) causes ~isUnlocked(W).

% Query 1: Try to open w1 when it is explicitly not a window, closed, and not unlocked
% (should be executable; after the action it stays closed and becomes not open) (satisfiable)
:- query
    0: ~isWindow(w1) & isClosed(w1) & ~isUnlocked(w1) & isOpen(w1);
    0: openWin(w1);
    maxstep: isClosed(w1) & ~isOpen(w1).

% Query 2: Try to open w1 when it is unlocked (opening is nonexecutable if isUnlocked is true) (unsatisfiable)
:- query
    0: ~isWindow(w1) & isClosed(w1) & isUnlocked(w1) & ~isOpen(w1);
    0: openWin(w1).

% Query 3: Try to close w1 when it is explicitly not a window and not open
% (should be executable; after the action it becomes open and not closed) (satisfiable)
:- query
    0: ~isWindow(w1) & ~isOpen(w1) & isClosed(w1);
    0: closeWin(w1);
    maxstep: isOpen(w1) & ~isClosed(w1).

% Query 4: Try to lock w1 when it is explicitly not a window, closed, and not unlocked
% (should be executable; after the action it becomes unlocked and not locked) (satisfiable)
:- query
    0: ~isWindow(w1) & isClosed(w1) & ~isUnlocked(w1) & isLocked(w1);
    0: lockWin(w1);
    maxstep: isUnlocked(w1) & ~isLocked(w1).

% Query 5: Try to unlock w1 when it is locked (unlocking is nonexecutable if isLocked is true) (unsatisfiable)

    0: unlockWin(w1).
