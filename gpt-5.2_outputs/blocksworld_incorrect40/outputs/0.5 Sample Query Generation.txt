% Query 1: Pick up a clear block that is not on the table while the hand is empty (satisfiable)
:- query
    0: handEmpty
       & loc(b5)=table
       & loc(b4)=b5 & clear(b4)
       & loc(b1)=table & clear(b1)
       & loc(b7)=table & clear(b7)
       & loc(b3)=table & clear(b3)
       & loc(b6)=table
       & loc(b2)=b6 & clear(b2);
    0: pickup(b4).

% Query 2: Try to pick up a block that is on the table (should be disallowed by nonexecutable pickup(B) if loc(B)=table) (unsatisfiable)
:- query
    0: handEmpty
       & loc(b5)=table
       & loc(b4)=b5 & clear(b4)
       & loc(b1)=table & clear(b1)
       & loc(b7)=table & clear(b7)
       & loc(b3)=table & clear(b3)
       & loc(b6)=table
       & loc(b2)=b6 & clear(b2);
    0: pickup(b1).

% Query 3: Try to put down a block that is not being held (should be disallowed by nonexecutable putdown(B) if ~holding(B)) (unsatisfiable)
:- query
    0: handEmpty
       & loc(b5)=table
       & loc(b4)=b5 & clear(b4)
       & loc(b1)=table & clear(b1)
       & loc(b7)=table & clear(b7)
       & loc(b3)=table & clear(b3)
       & loc(b6)=table
       & loc(b2)=b6 & clear(b2);
    0: putdown(b4).

% Query 4: Pick up b4 and then stack it onto clear block b1 (satisfiable)
:- query
    0: handEmpty
       & loc(b5)=table
       & loc(b4)=b5 & clear(b4)
       & loc(b1)=table & clear(b1)
       & loc(b7)=table & clear(b7)
       & loc(b3)=table & clear(b3)
       & loc(b6)=table
       & loc(b2)=b6 & clear(b2);
    0: pickup(b4);
    1: stack(b4,b1);
    maxstep: loc(b4)=b1 & handEmpty.

% Query 5: Try to stack b4 onto b5 even though b5 is not clear (b4 is on b5, so b5 cannot be clear) (unsatisfiable)
:- query
    0: handEmpty
       & loc(b5)=table
       & loc(b4)=b5 & clear(b4)
       & loc(b1)=table & clear(b1)
       & loc(b7)=table & clear(b7)
       & loc(b3)=table & clear(b3)
       & loc(b6)=table
       & loc(b2)=b6 & clear(b2);
    0: pickup(b4);
    1: stack(b4,b5).
