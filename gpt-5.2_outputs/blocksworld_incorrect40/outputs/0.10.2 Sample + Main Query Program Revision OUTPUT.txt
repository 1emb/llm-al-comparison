% PROGRAM CHANGED? [CHANGED]
% BC+ signature
:- sorts
    loc >> block.

:- objects
    b1, b2, b3, b4, b5, b6, b7 :: block;
    table :: loc.

:- variables
    B, B1, B2, B3, B4 :: block;
    L, L1, L2 :: loc.

:- constants
    pickup(block) :: exogenousAction;
    putdown(block) :: exogenousAction;
    stack(block, block) :: exogenousAction;
    unstack(block, block) :: exogenousAction;

    loc(block) :: inertialFluent(loc);
    clear(block) :: inertialFluent(boolean);
    holding(block) :: inertialFluent(boolean);
    handEmpty :: inertialFluent(boolean).

% Generated constraints/rules

% If some block is on B2, then B2 is not clear.
~clear(B2) if loc(B1)=B2.

% It is impossible for B2 to be clear while some block is on it.
impossible clear(B2) & loc(B1)=B2.

% It is impossible for a block to be located on itself.
impossible loc(B)=B.

% It is impossible for two different blocks to be held at the same time.
impossible holding(B1) & holding(B2) & B1\=B2.

% handEmpty and holding are mutually exclusive.
impossible handEmpty & holding(B).

% If the hand is empty, then it is holding no block.
~holding(B) if handEmpty.

% If the hand is holding some block, then the hand is not empty.
~handEmpty if holding(B).

% Picking up a block causes the hand to hold that block.
pickup(B) causes holding(B).

% Picking up a block causes the hand to not be empty.
pickup(B) causes ~handEmpty.

% IMPORTANT: represent "in hand" by setting loc(B)=table (no special hand location in this model).
pickup(B) causes loc(B)=table.

% Picking up a block causes the block to be clear (nothing on top while held).
pickup(B) causes clear(B).

% Picking up a block is not executable if the block is not clear.
nonexecutable pickup(B) if ~clear(B).

% Picking up a block is not executable if the hand is not empty.
nonexecutable pickup(B) if ~handEmpty.

% Picking up a block is not executable if the block is located on the table.
nonexecutable pickup(B) if loc(B)=table.

% Putting down a block causes the hand to no longer hold that block.
putdown(B) causes ~holding(B).

% Putting down a block causes the hand to become empty.
putdown(B) causes handEmpty.

% Putting down a block causes the block to be located on the table.
putdown(B) causes loc(B)=table.

% Putting down a block causes the block to be clear (nothing on top immediately after putdown).
putdown(B) causes clear(B).

% Putting down a block is not executable if the hand is not holding that block.
nonexecutable putdown(B) if ~holding(B).

% Stacking a block onto another block causes the hand to no longer hold the moved block.
stack(B1,B2) causes ~holding(B1).

% Stacking a block onto another block causes the hand to become empty.
stack(B1,B2) causes handEmpty.

% Stacking a block onto another block causes the moved block to be located on the destination block.
stack(B1,B2) causes loc(B1)=B2.

% Stacking makes the destination block not clear.
stack(B1,B2) causes ~clear(B2).

% After stacking, the moved block is clear.
stack(B1,B2) causes clear(B1).

% Stacking is not executable if the hand is not holding the moved block.
nonexecutable stack(B1,B2) if ~holding(B1).

% Stacking is not executable if the destination block is not clear.
nonexecutable stack(B1,B2) if ~clear(B2).

% Stacking is not executable if the moved block and the destination block are the same block.
nonexecutable stack(B1,B2) if B1=B2.

% Unstacking a block from on top of another block causes the hand to hold the first block.
unstack(B1,B2) causes holding(B1).

% Unstacking a block from on top of another block causes the hand to not be empty.
unstack(B1,B2) causes ~handEmpty.

% Unstacking causes the second block to become clear.
unstack(B1,B2) causes clear(B2).

% FIX: after unstacking, the moved block is NOT clear (it is being held).
unstack(B1,B2) causes ~clear(B1).

% Unstacking is not executable if the first block is not located on the second block.
nonexecutable unstack(B1,B2) if loc(B1)\=B2.

% Unstacking is not executable if the first block is not clear.
nonexecutable unstack(B1,B2) if ~clear(B1).

% Unstacking is not executable if the hand is not empty.
nonexecutable unstack(B1,B2) if ~handEmpty.

% No concurrency between these action pairs.
nonexecutable pickup(B1) & putdown(B2) if true.
nonexecutable pickup(B1) & stack(B2,B3) if true.
nonexecutable pickup(B1) & unstack(B2,B3) if true.
nonexecutable putdown(B1) & stack(B2,B3) if true.
nonexecutable putdown(B1) & unstack(B2,B3) if true.
nonexecutable stack(B1,B2) & unstack(B3,B4) if true.

% [FIX] Missing concurrency restriction: putdown and putdown cannot occur together.
% Without this, the solver can "putdown" multiple blocks in one step, which is impossible with one hand.
nonexecutable putdown(B1) & putdown(B2) if B1\=B2.

% [FIX] Missing concurrency restriction: pickup and pickup cannot occur together.
nonexecutable pickup(B1) & pickup(B2) if B1\=B2.

% [FIX] Missing concurrency restriction: stack and stack cannot occur together.
nonexecutable stack(B1,B2) & stack(B3,B4) if (B1\=B3 | B2\=B4).

% [FIX] Missing concurrency restriction: unstack and unstack cannot occur together.
nonexecutable unstack(B1,B2) & unstack(B3,B4) if (B1\=B3 | B2\=B4).

:- query

       & loc(b5)=table
       & loc(b4)=b5 & clear(b4)
