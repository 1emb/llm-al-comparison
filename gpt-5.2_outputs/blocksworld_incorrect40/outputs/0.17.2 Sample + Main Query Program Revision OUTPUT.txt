% It is impossible for a block to be located on itself.
impossible loc(B)=B.

% It is impossible for two different blocks to be held at the same time.
impossible holding(B1) & holding(B2) & B1\=B2.

% handEmpty and holding are mutually exclusive.

% If the hand is empty, then it is holding no block.
~holding(B) if handEmpty.

% If the hand is holding some block, then the hand is not empty.
~handEmpty if holding(B).

% Use a dedicated location "hand" to represent "in hand".

holding(B) if loc(B)=hand.
impossible holding(B) & loc(B)\=hand.

% Picking up a block causes the hand to hold that block.

% Picking up a block causes the hand to not be empty.
pickup(B) causes ~handEmpty.

% Picking up a block causes the block to be in hand.
pickup(B) causes loc(B)=hand.

% Picking up a block causes the block to be clear (nothing on top while held).
pickup(B) causes clear(B).

% Picking up a block is not executable if the hand is not empty.
nonexecutable pickup(B) if ~handEmpty.

% Picking up is not executable if the block is already in hand.

% Putting down a block causes the hand to no longer hold that block.
putdown(B) causes ~holding(B).

% Putting down a block causes the hand to become empty.
putdown(B) causes handEmpty.

% Putting down a block causes the block to be located on the table.
putdown(B) causes loc(B)=table.

putdown(B) causes clear(B).

nonexecutable putdown(B) if ~holding(B).

% Stacking a block onto another block causes the hand to no longer hold the moved block.
stack(B1,B2) causes ~holding(B1).

% Stacking a block onto another block causes the hand to become empty.
stack(B1,B2) causes handEmpty.

% Stacking a block onto another block causes the moved block to be located on the destination block.
stack(B1,B2) causes loc(B1)=B2.

% Stacking makes the destination block not clear.
stack(B1,B2) causes ~clear(B2).

% After stacking, the moved block is clear.

% Stacking is not executable if the hand is not holding the moved block.

% Stacking is not executable if the destination block is not clear.
nonexecutable stack(B1,B2) if ~clear(B2).

% Stacking is not executable if the moved block and the destination block are the same block.
nonexecutable stack(B1,B2) if B1=B2.

% Unstacking a block from on top of another block causes the hand to hold the first block.
unstack(B1,B2) causes holding(B1).

% Unstacking a block from on top of another block causes the hand to not be empty.
unstack(B1,B2) causes ~handEmpty.

% Unstacking causes the second block to become clear.

% Unstacking is not executable if the first block is not located on the second block.

nonexecutable unstack(B1,B2) if ~handEmpty.

% Unstacking is not executable if the moved block is already in hand.
