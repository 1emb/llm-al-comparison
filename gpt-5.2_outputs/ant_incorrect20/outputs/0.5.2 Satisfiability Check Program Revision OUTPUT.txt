% BC+ signature
:- sorts
    ant;
    place.

:- objects
    ant1 :: ant;
    p1, p2 :: place.

:- variables
    A :: ant;
    P, P1, P2, P3, P4 :: place.

:- constants
    % actions
    move(ant, place, place) :: exogenousAction;
    pickUpFood(ant, place) :: exogenousAction;
    dropFoodAtNest(ant, place) :: exogenousAction;
    clearObstacle(ant, place, place) :: exogenousAction;

    % fluents
    loc(ant) :: inertialFluent(place);
    clear(place) :: inertialFluent(boolean);
    hasFood(place) :: inertialFluent(boolean);
    carryingFood(ant) :: inertialFluent(boolean);

    % "rigid" relations encoded as inertial fluents (since rigidFluent is not supported here)
    adjacent(place, place) :: inertialFluent(boolean);
    isNest(place) :: inertialFluent(boolean).

% Generated constraints

% 1. Moving an ant from a starting place to a destination place makes the ant's current place become the destination place.
move(A,P1,P2) causes loc(A)=P2.

% 2. Moving is not executable unless the ant is currently at the starting place.
nonexecutable move(A,P1,P2) if loc(A)\=P1.

% 3. Moving is not executable if the starting place is adjacent to the destination place.
nonexecutable move(A,P1,P2) if adjacent(P1,P2).

% 4. Moving is not executable if the destination place is not clear.
nonexecutable move(A,P1,P2) if ~clear(P2).

% 5. Moving is not executable if the starting place and destination place are the same place.
nonexecutable move(A,P1,P2) if P1=P2.

% 6. Picking up food makes the ant carry food.
pickUpFood(A,P) causes carryingFood(A).

% 7. Picking up food makes that place no longer have food.
pickUpFood(A,P) causes ~hasFood(P).

% 8. Picking up food is not executable unless the ant is currently at that place.
nonexecutable pickUpFood(A,P) if loc(A)\=P.

% 9. Picking up food is not executable if that place does not have food.
nonexecutable pickUpFood(A,P) if ~hasFood(P).

% 10. Picking up food is not executable if the ant is already carrying food.
nonexecutable pickUpFood(A,P) if carryingFood(A).

% 11. Dropping food at a nest makes the ant no longer carry food.
dropFoodAtNest(A,P) causes ~carryingFood(A).

% 12. Dropping food at a nest makes that place have food.
dropFoodAtNest(A,P) causes hasFood(P).

% 13. Dropping food at a nest is not executable unless the ant is currently at that place.
nonexecutable dropFoodAtNest(A,P) if loc(A)\=P.

% 14. Dropping food at a nest is not executable if that place is not a nest.
nonexecutable dropFoodAtNest(A,P) if ~isNest(P).

% 15. Dropping food at a nest is not executable if the ant is not carrying food.
nonexecutable dropFoodAtNest(A,P) if ~carryingFood(A).

% 16. Clearing an obstacle makes the blocked place clear.
clearObstacle(A,P1,P2) causes clear(P2).

% 17. Clearing an obstacle is not executable unless the ant is currently at the first place.
nonexecutable clearObstacle(A,P1,P2) if loc(A)\=P1.

% 18. Clearing an obstacle is not executable if the first place is adjacent to the blocked place.
nonexecutable clearObstacle(A,P1,P2) if adjacent(P1,P2).

% 19. Clearing an obstacle is not executable if the blocked place is already clear.
nonexecutable clearObstacle(A,P1,P2) if clear(P2).

% 20. Clearing an obstacle is not executable if the first place and the blocked place are the same place.
nonexecutable clearObstacle(A,P1,P2) if P1=P2.

% 21-26. No concurrent actions by the same ant in the same step (pairwise).
nonexecutable move(A,P1,P2) & pickUpFood(A,P3).
nonexecutable move(A,P1,P2) & dropFoodAtNest(A,P3).
nonexecutable move(A,P1,P2) & clearObstacle(A,P3,P4).
nonexecutable pickUpFood(A,P1) & dropFoodAtNest(A,P2).
nonexecutable pickUpFood(A,P1) & clearObstacle(A,P2,P3).
nonexecutable dropFoodAtNest(A,P1) & clearObstacle(A,P2,P3).

% Main query
% Original natural language query:
% Ant ant1 is at p1. p1 is adjacent to p2 and p2 is adjacent to p1.
% p1 is clear and p2 is clear. p1 is the nest. Food is present at p2.
% Find a plan to bring the food to the nest.
:- query
    0: loc(ant1)=p1 & adjacent(p1,p2) & adjacent(p2,p1) & clear(p1) & clear(p2) & isNest(p1) & hasFood(p2) & ~carryingFood(ant1);
    maxstep: hasFood(p1).
