:- sorts
    window.


:- objects
    w1, w2 :: window.


:- variables
    W, W1, W2, W3, W4 :: window;
    Bool, Bool1, Bool2, Bool3, Bool4 :: boolean.


:- constants
    openWin(window) :: exogenousAction;
    closeWin(window) :: exogenousAction;
    lockWin(window) :: exogenousAction;
    unlockWin(window) :: exogenousAction;

    isOpen(window) :: inertialFluent(boolean);
    isClosed(window) :: inertialFluent(boolean);
    isLocked(window) :: inertialFluent(boolean);
    isUnlocked(window) :: inertialFluent(boolean).

% Generated constraints
% 1. A window cannot be both open and closed at the same time.
impossible isOpen(W) & isClosed(W).

% 2. A window cannot be both locked and unlocked at the same time.
impossible isLocked(W) & isUnlocked(W).

% 3. Opening a window makes that window open.
openWin(W) causes isOpen(W).

% 4. Opening a window makes that window not closed.
openWin(W) causes ~isClosed(W).

% 5. Opening a window is not executable if the window is already open.
nonexecutable openWin(W) if isOpen(W).

% 6. Opening a window is not executable if the window is not closed.
nonexecutable openWin(W) if ~isClosed(W).

% 7. Opening a window is not executable if the window is not unlocked.
nonexecutable openWin(W) if ~isUnlocked(W).

% 8. Closing a window makes that window closed.
closeWin(W) causes isClosed(W).

% 9. Closing a window makes that window not open.
closeWin(W) causes ~isOpen(W).

% 10. Closing a window is not executable if the window is already closed.
nonexecutable closeWin(W) if isClosed(W).

% 11. Closing a window is not executable if the window is not open.
nonexecutable closeWin(W) if ~isOpen(W).

% 12. Locking a window makes that window locked.
lockWin(W) causes isLocked(W).

% 13. Locking a window makes that window not unlocked.
lockWin(W) causes ~isUnlocked(W).

% 14. Locking a window is not executable if the window is already locked.
nonexecutable lockWin(W) if isLocked(W).

% 15. Locking a window is not executable if the window is not closed.
nonexecutable lockWin(W) if ~isClosed(W).

% 16. Locking a window is not executable if the window is not unlocked.
nonexecutable lockWin(W) if ~isUnlocked(W).

% 17. Unlocking a window makes that window unlocked.
unlockWin(W) causes isUnlocked(W).

% 18. Unlocking a window makes that window not locked.
unlockWin(W) causes ~isLocked(W).

% 19. Unlocking a window is not executable if the window is already unlocked.
nonexecutable unlockWin(W) if isUnlocked(W).

% 20. Unlocking a window is not executable if the window is not locked.
nonexecutable unlockWin(W) if ~isLocked(W).


noconcurrency.

:- query
    0: isClosed(w1) & isUnlocked(w1) & isOpen(w2) & isUnlocked(w2);
    maxstep: isOpen(w1) & isLocked(w2).
