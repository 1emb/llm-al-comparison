% PROGRAM CHANGED? [CHANGED]
% BC+ signature
:- sorts
    loc >> block.

:- objects
    b1, b2, b3, b4, b5, b6, b7 :: block;
    table :: loc.

:- variables
    B, B1, B2, B3, B4 :: block;
    L, L1, L2, L3, L4 :: loc.

:- constants
    pickup(block) :: exogenousAction;
    putdown(block) :: exogenousAction;
    stack(block, block) :: exogenousAction;
    unstack(block, block) :: exogenousAction;

    loc(block) :: inertialFluent(loc);
    clear(block) :: inertialFluent(boolean);
    onTable(block) :: inertialFluent(boolean);
    handEmpty :: inertialFluent(boolean);
    holding(block) :: inertialFluent(boolean).

% Generated constraints

% 1. A block cannot be located on itself.
impossible loc(B)=B.

% 2. A block cannot be located on two different locations at the same time.
impossible loc(B)=L1 & loc(B)=L2 & L1\=L2.

% 3. If loc(B)=B1 (a block), then B is not on the table.
~onTable(B) if loc(B)=B1 & B1\=table.

% 4. If onTable(B) then loc(B)=table.
loc(B)=table if onTable(B).

% 5. If loc(B)=B1 (a block), then B1 is not clear.
~clear(B1) if loc(B)=B1 & B1\=table.

% 6. If clear(B1) then no block has location B1.
impossible clear(B1) & loc(B)=B1.

% 7. Clear should be equivalent to "nothing on top".
% If B1 is not clear, then some block is on B1.
% (Fix) Existence encoding: if B1 is not clear, then there exists some B with loc(B)=B1.
% In BC+ we encode this as: it is impossible that B1 is not clear AND for all blocks B, loc(B) is not B1.
% Grounding trick: for each B, forbid (~clear(B1) & loc(B)\=B1).
impossible ~clear(B1) & loc(B)\=B1.

% 8. The hand cannot be empty and holding a block at the same time.
impossible handEmpty & holding(B).

% 9. If holding(B) then handEmpty is not the case.
~handEmpty if holding(B).

% 10. If handEmpty then holding(B) is not the case for every block B.
~holding(B) if handEmpty.

% 11. The hand cannot be holding two different blocks at the same time.
impossible holding(B1) & holding(B2) & B1\=B2.

% --- Action effects and executability conditions ---
% Standard blocks-world meaning:
% - pickup: pick up from table
% - putdown: put onto table
% - unstack: take top block off bottom block
% - stack: put held block onto clear block

% pickup(B): allowed only if B is clear, B IS on the table, and hand is empty.
% Effects: holding(B), hand not empty, B not on table.
pickup(B) causes holding(B).
pickup(B) causes ~handEmpty.
pickup(B) causes ~onTable(B).

nonexecutable pickup(B) if ~clear(B).
nonexecutable pickup(B) if ~onTable(B).
nonexecutable pickup(B) if ~handEmpty.

% putdown(B): allowed only if holding(B).
% Effects: B on table, hand empty, not holding(B).
putdown(B) causes onTable(B).
putdown(B) causes handEmpty.
putdown(B) causes ~holding(B).

nonexecutable putdown(B) if ~holding(B).

% stack(B1,B2): allowed only if holding(B1) and B2 is clear and B1!=B2.
% Effects: loc(B1)=B2, hand empty, not holding(B1).
stack(B1,B2) causes loc(B1)=B2.
stack(B1,B2) causes handEmpty.
stack(B1,B2) causes ~holding(B1).

nonexecutable stack(B1,B2) if ~holding(B1).
nonexecutable stack(B1,B2) if ~clear(B2).
nonexecutable stack(B1,B2) if B1=B2.

% unstack(B1,B2): allowed only if loc(B1)=B2, B1 is clear, and hand is empty.
% Effects: holding(B1), hand not empty, and B1 is no longer on B2 (so put it on table).
unstack(B1,B2) causes holding(B1).
unstack(B1,B2) causes ~handEmpty.
unstack(B1,B2) causes loc(B1)=table.

nonexecutable unstack(B1,B2) if loc(B1)\=B2.
nonexecutable unstack(B1,B2) if ~clear(B1).
nonexecutable unstack(B1,B2) if ~handEmpty.
nonexecutable unstack(B1,B2) if B1=B2.

% Concurrency restrictions (kept)
nonexecutable pickup(B1) & putdown(B2) if true.
nonexecutable pickup(B1) & stack(B2,B3) if true.
nonexecutable pickup(B1) & unstack(B2,B3) if true.
nonexecutable putdown(B1) & stack(B2,B3) if true.
nonexecutable putdown(B1) & unstack(B2,B3) if true.
nonexecutable stack(B1,B2) & unstack(B3,B4) if true.


% MAIN QUERY CHANGED? [UNCHANGED]
:- query
    0: handEmpty
       & onTable(b5) & loc(b5)=table & ~clear(b5)
       & loc(b1)=b5 & clear(b1) & ~onTable(b1)
       & onTable(b3) & loc(b3)=table & clear(b3)
       & onTable(b2) & loc(b2)=table & ~clear(b2)
       & loc(b4)=b2 & clear(b4) & ~onTable(b4);
    maxstep: onTable(b3) & loc(b3)=table & clear(b3)
             & onTable(b1) & loc(b1)=table
             & loc(b4)=b1 & clear(b4)
             & onTable(b2) & loc(b2)=table & clear(b2)
             & onTable(b5) & loc(b5)=table & clear(b5).

% Query 1: Unstack a clear block b1 from b5 while hand is empty (satisfiable)
:- query
    0: handEmpty
       & loc(b1)=b5 & ~onTable(b1) & clear(b1)
       & onTable(b5) & loc(b5)=table & ~clear(b5)
       & onTable(b2) & loc(b2)=table & clear(b2);
    0: unstack(b1,b5).

% Query 2: Pick up a block that is on the table (should be allowed if clear and hand empty) (satisfiable)
:- query
    0: handEmpty
       & onTable(b3) & loc(b3)=table & clear(b3)
       & onTable(b5) & loc(b5)=table & clear(b5);
    0: pickup(b3).

% Query 3: Put down a block that is not being held (unsatisfiable)
:- query
    0: handEmpty
       & ~holding(b4)
       & onTable(b2) & loc(b2)=table & clear(b2);
    0: putdown(b4).

% Query 4: Stack onto a destination block that is not clear (unsatisfiable)
:- query
    0: ~handEmpty & holding(b1)
       & ~onTable(b1) & clear(b1)
       & loc(b4)=b2 & ~onTable(b4) & clear(b4)
       & loc(b2)=table & onTable(b2) & ~clear(b2);
    0: stack(b1,b2).

% Query 5: Unstack a clear top block from a bottom block while the hand is empty (satisfiable)
:- query
    0: handEmpty
       & loc(b4)=b2 & ~onTable(b4) & clear(b4)
       & loc(b2)=table & onTable(b2) & ~clear(b2)
       & onTable(b5) & loc(b5)=table & clear(b5);
    0: unstack(b4,b2).
