% PROGRAM CHANGED? [CHANGED]
% BC+ signature
:- sorts
    entity.

:- objects
    a, b, c, d :: entity.

:- variables
    E, E1, E2 :: entity.

:- constants
    attack(entity) :: exogenousAction;
    succumb(entity) :: exogenousAction;
    overcome(entity, entity) :: exogenousAction;
    feast(entity, entity) :: exogenousAction;

    isProvince(entity) :: inertialFluent(boolean);
    isPlanet(entity) :: inertialFluent(boolean);
    harmony :: inertialFluent(boolean);
    inPain(entity) :: inertialFluent(boolean);
    craves(entity, entity) :: inertialFluent(boolean).

% Generated constraints

% Attack preconditions
nonexecutable attack(E) if ~isProvince(E).
nonexecutable attack(E) if isPlanet(E).
nonexecutable attack(E) if harmony.

% Attack effects
attack(E) causes inPain(E).
attack(E) causes ~isProvince(E).
attack(E) causes isPlanet(E).
attack(E) causes harmony.

% Succumb preconditions
nonexecutable succumb(E) if inPain(E).

% Succumb effects (per problem statement)
succumb(E) causes isProvince(E).
succumb(E) causes ~harmony.
succumb(E) causes isPlanet(E).
succumb(E) causes inPain(E).

% Overcome preconditions
nonexecutable overcome(E1,E2) if isProvince(E2).
nonexecutable overcome(E1,E2) if ~inPain(E1).

% Overcome effects
overcome(E1,E2) causes ~harmony.
overcome(E1,E2) causes isProvince(E1).
overcome(E1,E2) causes ~craves(E1,E2).
overcome(E1,E2) causes ~isProvince(E2).
overcome(E1,E2) causes inPain(E1).

% Feast preconditions
nonexecutable feast(E1,E2) if craves(E1,E2).
nonexecutable feast(E1,E2) if ~isProvince(E1).
nonexecutable feast(E1,E2) if harmony.

% Feast effects (per problem statement)
feast(E1,E2) causes ~inPain(E1).
feast(E1,E2) causes ~isProvince(E2).
feast(E1,E2) causes craves(E1,E2).
feast(E1,E2) causes ~isProvince(E1).
feast(E1,E2) causes ~harmony.

% Enforce intended exclusivity between province and planet
impossible isProvince(E) & isPlanet(E).

% [CHANGED] Remove these "impossible" constraints: they are not valid in BC+ because
% they contain negation (~) inside an "impossible ..." fluent formula.
% (Also, boolean fluents are already 2-valued in BC+/Cplus2ASP; you do not need to
% enforce "not both true and false" this way.)
% impossible inPain(E) & ~inPain(E).
% impossible harmony & ~harmony.
% impossible craves(E1,E2) & ~craves(E1,E2).

% MAIN QUERY CHANGED? [UNCHANGED]
:- query
    0: harmony
       & craves(a,c) & craves(b,a) & craves(c,d)
       & ~craves(a,a) & ~craves(a,b) & ~craves(a,d)
       & ~craves(b,b) & ~craves(b,c) & ~craves(b,d)
       & ~craves(c,a) & ~craves(c,b) & ~craves(c,c)
       & ~craves(d,a) & ~craves(d,b) & ~craves(d,c) & ~craves(d,d)
       & isPlanet(d)
       & ~isPlanet(a) & ~isPlanet(b) & ~isPlanet(c)
       & isProvince(b)
       & ~isProvince(a) & ~isProvince(c) & ~isProvince(d)
       & ~inPain(a) & ~inPain(b) & ~inPain(c) & ~inPain(d);
    maxstep: craves(b,c) & craves(d,b).

% SAMPLE QUERIES CHANGED? [CHANGED]
% Query 1: Attack is blocked when harmony is present (unsatisfiable)
:- query
    0: harmony & isProvince(b) & ~isPlanet(b);
    0: attack(b).

% Query 2: Attack is executable when target is a province, not a planet, and harmony is absent; after attack, effects hold (satisfiable)
:- query
    0: ~harmony & isProvince(b) & ~isPlanet(b) & ~inPain(b);
    0: attack(b);
    maxstep: inPain(b) & ~isProvince(b) & isPlanet(b) & harmony.

% Query 3: Succumb is blocked if the entity is already in pain (unsatisfiable)
:- query
    0: inPain(a);
    0: succumb(a).

% Query 4: Overcome is blocked if the first entity is not in pain (unsatisfiable)
:- query
    0: ~inPain(a) & ~isProvince(b);
    0: overcome(a,b).

% [CHANGED] Query 5 initial state must not contradict "impossible isProvince(E) & isPlanet(E)".
% Feast is executable when the first entity is a province, does not crave the second, and harmony is absent; after feast, effects hold (satisfiable)
:- query
    0: ~harmony & isProvince(a) & ~isPlanet(a) & ~craves(a,b) & inPain(a) & isProvince(b) & ~isPlanet(b);
    0: feast(a,b);
    maxstep: ~inPain(a) & craves(a,b) & ~isProvince(a) & ~isProvince(b) & ~harmony.
