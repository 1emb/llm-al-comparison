% Generated constraints

% Boarding preconditions: must be at a floor, lift at that floor, and board only at origin.
nonexecutable board(P,F) if ~isFloor(F).
nonexecutable board(P,F) if ~liftAt(F).
nonexecutable board(P,F) if isPassenger(P).
nonexecutable board(P,F) if origin(P)\=F.

% Boarding effect: person becomes boarded.
board(P,F) causes boarded(P).

% Departing preconditions: must be at a floor, lift at that floor, and depart only at destination, and must already be boarded.
nonexecutable depart(P,F) if ~isFloor(F).
nonexecutable depart(P,F) if ~liftAt(F).
nonexecutable depart(P,F) if isPassenger(P).
nonexecutable depart(P,F) if dest(P)\=F.
nonexecutable depart(P,F) if ~boarded(P).

% Departing effects: person becomes unboarded and served.
depart(P,F) causes ~boarded(P).
depart(P,F) causes served(P).

% Movement: endpoints are floors, lift at source, and above relation matches direction.

% Move up: from lower floor F1 to higher floor F2, so above(F2,F1) must hold.
nonexecutable moveUp(F1,F2) if ~isFloor(F1).
nonexecutable moveUp(F1,F2) if ~isFloor(F2).
nonexecutable moveUp(F1,F2) if ~liftAt(F1).
nonexecutable moveUp(F1,F2) if ~above(F2,F1).
moveUp(F1,F2) causes ~liftAt(F1).
moveUp(F1,F2) causes liftAt(F2).

% Move down: from higher floor F1 to lower floor F2, so above(F1,F2) must hold.
nonexecutable moveDown(F1,F2) if ~isFloor(F1).
nonexecutable moveDown(F1,F2) if ~isFloor(F2).
nonexecutable moveDown(F1,F2) if ~liftAt(F1).
nonexecutable moveDown(F1,F2) if ~above(F1,F2).
moveDown(F1,F2) causes ~liftAt(F1).
moveDown(F1,F2) causes liftAt(F2).

% Sanity constraint: lift cannot be at two floors at once.
impossible liftAt(F1) & liftAt(F2) & F1\=F2.
