% BC+ signature
:- sorts
    floor;
    passenger.


:- objects
    f0, f1, f2, f3 :: floor;
    p0, p1 :: passenger.


:- variables
    F, F1, F2 :: floor;
    P, P1 :: passenger.


:- constants
    board(passenger,floor) :: exogenousAction;
    depart(passenger,floor) :: exogenousAction;
    moveUp(floor,floor) :: exogenousAction;
    moveDown(floor,floor) :: exogenousAction;

    isFloor(floor) :: inertialFluent(boolean);
    isPassenger(passenger) :: inertialFluent(boolean);
    liftAt(floor) :: inertialFluent(boolean);
    above(floor,floor) :: inertialFluent(boolean);

    origin(passenger) :: inertialFluent(floor);
    dest(passenger) :: inertialFluent(floor);
    boarded(passenger) :: inertialFluent(boolean);
    served(passenger) :: inertialFluent(boolean).

% Generated constraints

% Boarding preconditions: must be at a floor, lift at that floor, person is not a passenger, and board only at origin.
nonexecutable board(P,F) if ~isFloor(F).
nonexecutable board(P,F) if ~liftAt(F).
nonexecutable board(P,F) if isPassenger(P).
nonexecutable board(P,F) if origin(P)\=F.

% Boarding effect: person becomes boarded.
board(P,F) causes boarded(P).

% Departing preconditions: must be at a floor, lift at that floor, person is not a passenger, depart only at destination, and must already be boarded.
nonexecutable depart(P,F) if ~isFloor(F).
nonexecutable depart(P,F) if ~liftAt(F).
nonexecutable depart(P,F) if isPassenger(P).
nonexecutable depart(P,F) if dest(P)\=F.
nonexecutable depart(P,F) if ~boarded(P).

% Departing effects: person becomes unboarded and served.
depart(P,F) causes ~boarded(P).
depart(P,F) causes served(P).

% Movement: endpoints are floors, lift at source, and above relation matches direction.

% Move up: from lower floor F1 to higher floor F2, so above(F2,F1) must hold.
nonexecutable moveUp(F1,F2) if ~isFloor(F1).
nonexecutable moveUp(F1,F2) if ~isFloor(F2).
nonexecutable moveUp(F1,F2) if ~liftAt(F1).
nonexecutable moveUp(F1,F2) if ~above(F2,F1).
moveUp(F1,F2) causes ~liftAt(F1).
moveUp(F1,F2) causes liftAt(F2).

% Move down: from higher floor F1 to lower floor F2, so above(F1,F2) must hold.
nonexecutable moveDown(F1,F2) if ~isFloor(F1).
nonexecutable moveDown(F1,F2) if ~isFloor(F2).
nonexecutable moveDown(F1,F2) if ~liftAt(F1).
nonexecutable moveDown(F1,F2) if ~above(F1,F2).
moveDown(F1,F2) causes ~liftAt(F1).
moveDown(F1,F2) causes liftAt(F2).

% Sanity constraint: lift cannot be at two floors at once.
impossible liftAt(F1) & liftAt(F2) & F1\=F2.

% [ADDED] Sanity constraint: a passenger cannot be both boarded and served.
impossible boarded(P) & served(P).

% [ADDED] Sanity constraint: served implies not boarded (redundant with the above, but clearer).
impossible served(P) & boarded(P).

% [ADDED] Sanity constraint: above should be irreflexive (prevents above(f,f) models seen in outputs).
impossible above(F,F).

% [ADDED] Sanity constraint: above should be antisymmetric (prevents cycles like above(f3,f0) together with above(f0,f3)).
impossible above(F1,F2) & above(F2,F1) & F1\=F2.
