% PROGRAM CHANGED? [CHANGED]
% BC+ signature
:- sorts
    floor;
    passenger.

:- objects
    f0,f1,f2,f3 :: floor;
    p0,p1 :: passenger.

:- variables
    F, F1, F2, F3, F4 :: floor;
    P, P1, P2, P3, P4 :: passenger.

:- constants
    board(floor, passenger) :: exogenousAction;
    depart(floor, passenger) :: exogenousAction;
    up(floor, floor) :: exogenousAction;
    down(floor, floor) :: exogenousAction;

    isFloor(floor) :: inertialFluent(boolean);
    isPassenger(passenger) :: inertialFluent(boolean);
    liftAt(floor) :: inertialFluent(boolean);

    % Treat "above" as fixed by making it inertial and never changing it.
    above(floor, floor) :: inertialFluent(boolean);

    origin(passenger) :: inertialFluent(floor);
    dest(passenger) :: inertialFluent(floor);
    boarded(passenger) :: inertialFluent(boolean);
    served(passenger) :: inertialFluent(boolean).

% Generated constraints

% 1. If a passenger is boarded, then they are served.
served(P) if boarded(P).

% 2. Each passenger's origin is a floor.
isFloor(F) if origin(P)=F.

% 3. Each passenger's destination is a floor.
isFloor(F) if dest(P)=F.

% 4. A passenger's origin and destination cannot be the same floor.
impossible origin(P)=F & dest(P)=F.

% 5. The lift cannot be at two different floors at the same time.
impossible liftAt(F1) & liftAt(F2) & F1\=F2.

% 6. board(F,P) is not executable if F is not a floor.
nonexecutable board(F,P) if ~isFloor(F).

% 7. board(F,P) is not executable if the lift is not at F.
nonexecutable board(F,P) if ~liftAt(F).

% 9. board(F,P) is not executable if P's origin is not F.
% Boarding should be allowed only at the origin floor.
nonexecutable board(F,P) if origin(P)\=F.

% 10. board(F,P) causes P to be boarded.
board(F,P) causes boarded(P).

% 11. depart(F,P) is not executable if F is not a floor.
nonexecutable depart(F,P) if ~isFloor(F).

% 12. depart(F,P) is not executable if the lift is not at F.
nonexecutable depart(F,P) if ~liftAt(F).

% 14. depart(F,P) is not executable if P's destination is not F.
% Departing should be allowed only at the destination floor.
nonexecutable depart(F,P) if dest(P)\=F.

% 15. depart(F,P) is not executable if P is not boarded.
% You can only depart if you are currently boarded.
nonexecutable depart(F,P) if ~boarded(P).

% 16. depart(F,P) causes P to not be boarded.
depart(F,P) causes ~boarded(P).

% 17. depart(F,P) causes P to be served.
depart(F,P) causes served(P).

% 18. up(F1,F2) is not executable if F1 is not a floor.
nonexecutable up(F1,F2) if ~isFloor(F1).

% 19. up(F1,F2) is not executable if F2 is not a floor.
nonexecutable up(F1,F2) if ~isFloor(F2).

% 20. up(F1,F2) is not executable if the lift is not at F1.
nonexecutable up(F1,F2) if ~liftAt(F1).

% 21. up(F1,F2) is not executable if F2 is not above F1.
nonexecutable up(F1,F2) if ~above(F2,F1).

% 22. up(F1,F2) causes the lift to be at F2.
up(F1,F2) causes liftAt(F2).

% 23. up(F1,F2) causes the lift to not be at F1.
up(F1,F2) causes ~liftAt(F1).

% 24. down(F1,F2) is not executable if F1 is not a floor.
nonexecutable down(F1,F2) if ~isFloor(F1).

% 25. down(F1,F2) is not executable if F2 is not a floor.
nonexecutable down(F1,F2) if ~isFloor(F2).

% 26. down(F1,F2) is not executable if the lift is not at F1.
nonexecutable down(F1,F2) if ~liftAt(F1).

% 27. down(F1,F2) is not executable if F2 is above F1.
nonexecutable down(F1,F2) if above(F2,F1).

% 28. down(F1,F2) causes the lift to be at F2.
down(F1,F2) causes liftAt(F2).

% 29. down(F1,F2) causes the lift to not be at F1.
down(F1,F2) causes ~liftAt(F1).

% 30. It is not executable for the lift to move up and move down at the same time.
nonexecutable up(F1,F2) & down(F3,F4).

% 31. It is not executable to perform board(F,P) and depart(F,P) at the same time for the same F and P.
nonexecutable board(F,P) & depart(F,P).

% Helpful: Do not allow boarding if already boarded.
nonexecutable board(F,P) if boarded(P).

% Helpful: Do not allow departing if already served.
nonexecutable depart(F,P) if served(P).

% Do not allow boarding/departing concurrently with movement.
nonexecutable board(F,P) & up(F1,F2).
nonexecutable board(F,P) & down(F1,F2).
nonexecutable depart(F,P) & up(F1,F2).
nonexecutable depart(F,P) & down(F1,F2).

% [CHANGED]
% The sample/main outputs show the solver "inventing" above-relations like above(f2,f2), above(f3,f0), above(f3,f3).
% That happens because above/2 is inertial and unconstrained, so it can be arbitrarily set unless ruled out.
% Since "above" is intended to be a strict order, constrain it:
impossible above(F,F).
impossible above(F1,F2) & above(F2,F1) & F1\=F2.

% [CHANGED]
% The reason query q4 (and even the "last satisfiable state" you listed) becomes UNSAT is that
% boarded/1 and served/1 are inertialFluent(boolean) but the queries never set their values for
% all passengers at time 0. In BC+, leaving an inertial boolean fluent unspecified does NOT mean
% it is false; it can be chosen true, and then it can violate:
%   - nonexecutable board(F,P) if boarded(P).
%   - nonexecutable depart(F,P) if served(P).
% So the solver can pick boarded(p0)=true or served(p0)=true initially, making actions like down/2
% irrelevant but making the overall initial state inconsistent with later action constraints.
% Fix: provide defaults so that, unless stated otherwise, passengers start unboarded and unserved.
default ~boarded(P).
default ~served(P).

    0: above(f0,f1) & above(f0,f2) & above(f0,f3) &
       above(f1,f2) & above(f1,f3) &
       above(f2,f3);

    0: origin(p0)=f3 & dest(p0)=f0;

    0: liftAt(f0);

    0: ~boarded(p0) & ~served(p0) & ~boarded(p1) & ~served(p1);

    maxstep: served(p0) & served(p1).
