:- sorts
    vehicle >> truck;
    vehicle >> airplane;
    thing >> package;
    thing >> vehicle;
    package;
    truck;
    airplane;
    location;
    city.


:- objects
    % Domain objects are not specified in the problem statement.
    % They can be added later as needed, e.g.:
    % p1, p2 :: package;
    % t1, t2 :: truck;
    % a1, a2 :: airplane;
    % l1, l2 :: location;
    % c1, c2 :: city.


:- variables
    P, P1, P2, P3, P4 :: package;
    T, T1, T2, T3, T4 :: truck;
    A, A1, A2, A3, A4 :: airplane;
    V, V1, V2, V3, V4 :: vehicle;
    Th, Th1, Th2, Th3, Th4 :: thing;
    L, L1, L2, L3, L4 :: location;
    C, C1, C2, C3, C4 :: city;
    Bool, Bool1, Bool2, Bool3, Bool4 :: boolean.


:- constants
    loadTruck(package,truck,location) :: exogenousAction;
    loadAirplane(package,airplane,location) :: exogenousAction;
    unloadTruck(package,truck,location) :: exogenousAction;
    unloadAirplane(package,airplane,location) :: exogenousAction;
    drive(truck,location,location,city) :: exogenousAction;
    fly(airplane,location,location) :: exogenousAction;

    loc(thing) :: inertialFluent(location);
    in(package,vehicle) :: inertialFluent(boolean);
    inCity(location) :: inertialFluent(city);
    isAirport(location) :: inertialFluent(boolean).


% constraints/rules
% 1. A package cannot be inside two different vehicles at the same time.
impossible in(P,V1) & in(P,V2) & V1\=V2.

% 2. If a package is inside a vehicle, then the package is at the same location as that vehicle.
loc(P)=L if in(P,V) & loc(V)=L.

% 3. If a package is inside a vehicle, then the package is not directly at any location different from the vehicle's location.
impossible in(P,V) & loc(V)=L1 & loc(P)=L2 & L1\=L2.

% 4. Loading a package into a truck at a location causes the package to be inside that truck.
loadTruck(P,T,L) causes in(P,T).

% 5. Loading a package into a truck at a location causes the package to be at that location.
loadTruck(P,T,L) causes loc(P)=L.

% 6. Loading a package into a truck at a location is not executable if the package is not at that location.
nonexecutable loadTruck(P,T,L) if loc(P)\=L.

% 7. Loading a package into a truck at a location is not executable if the truck is not at that location.
nonexecutable loadTruck(P,T,L) if loc(T)\=L.

% 8. Loading a package into a truck at a location is not executable if the package is already inside some vehicle.
nonexecutable loadTruck(P,T,L) if in(P,V).

% 9. Loading a package into an airplane at a location causes the package to be inside that airplane.
loadAirplane(P,A,L) causes in(P,A).

% 10. Loading a package into an airplane at a location causes the package to not be directly at that location.
loadAirplane(P,A,L) causes loc(P)=L1 if loc(P)=L1 & L1\=L.

% 11. Loading a package into an airplane at a location is not executable if the airplane is not at that location.
nonexecutable loadAirplane(P,A,L) if loc(A)\=L.

% 12. Loading a package into an airplane at a location is not executable if the package is already inside some vehicle.
nonexecutable loadAirplane(P,A,L) if in(P,V).

% 13. Unloading a package from a truck at a location causes the package to not be inside that truck.
unloadTruck(P,T,L) causes ~in(P,T).

% 14. Unloading a package from a truck at a location causes the package to be at that location.
unloadTruck(P,T,L) causes loc(P)=L.

% 15. Unloading a package from a truck at a location is not executable if the truck is not at that location.
nonexecutable unloadTruck(P,T,L) if loc(T)\=L.

% 16. Unloading a package from a truck at a location is not executable if the package is not inside that truck.
nonexecutable unloadTruck(P,T,L) if ~in(P,T).

% 17. Unloading a package from an airplane at a location causes the package to not be inside that airplane.
unloadAirplane(P,A,L) causes ~in(P,A).

% 18. Unloading a package from an airplane at a location causes the package to be at that location.
unloadAirplane(P,A,L) causes loc(P)=L.

% 19. Unloading a package from an airplane at a location is not executable if the airplane is not at that location.
nonexecutable unloadAirplane(P,A,L) if loc(A)\=L.

% 20. Unloading a package from an airplane at a location is not executable if the package is not inside that airplane.
nonexecutable unloadAirplane(P,A,L) if ~in(P,A).

% 21. Driving a truck from one location to another in a city causes the truck to be at the to-location.
drive(T,L1,L2,C) causes loc(T)=L2.

% 22. Driving a truck from one location to another in a city causes the truck to not be at the from-location.
drive(T,L1,L2,C) causes loc(T)=L if loc(T)=L & L\=L2.

% 23. Driving a truck from one location to another in a city is not executable if the truck is not at the from-location.
nonexecutable drive(T,L1,L2,C) if loc(T)\=L1.

% 24. Driving a truck from one location to another in a city is not executable if the from-location is not in that city.
nonexecutable drive(T,L1,L2,C) if inCity(L1)\=C.

% 25. Driving a truck from one location to another in a city is not executable if the to-location is not in that city.
nonexecutable drive(T,L1,L2,C) if inCity(L2)\=C.

% 26. Driving a truck from one location to another in a city is not executable if the from-location is the same as the to-location.
nonexecutable drive(T,L1,L2,C) if L1=L2.

% 27. Flying an airplane from one location to another causes the airplane to be at the to-location.
fly(A,L1,L2) causes loc(A)=L2.

% 28. Flying an airplane from one location to another causes the airplane to not be at the from-location.
fly(A,L1,L2) causes loc(A)=L if loc(A)=L & L\=L2.

% 29. Flying an airplane from one location to another is not executable if the airplane is not at the from-location.
nonexecutable fly(A,L1,L2) if loc(A)\=L1.

% 30. Flying an airplane from one location to another is not executable if the from-location is not an airport.
nonexecutable fly(A,L1,L2) if ~isAirport(L1).

% 31. Flying an airplane from one location to another is not executable if the to-location is not an airport.
nonexecutable fly(A,L1,L2) if ~isAirport(L2).

% 32. Flying an airplane from one location to another is not executable if the from-location is the same as the to-location.
nonexecutable fly(A,L1,L2) if L1=L2.

% 33. If a truck drives, every package inside that truck moves with it (locations match after the move).
loc(P)=L2 if in(P,T) after drive(T,L1,L2,C).

% 34. If an airplane flies, every package inside that airplane moves with it (locations match after the move).
loc(P)=L2 if in(P,A) after fly(A,L1,L2).

% 35. It is not executable to load and unload the same package with the same truck at the same location at the same time.
nonexecutable loadTruck(P,T,L) & unloadTruck(P,T,L) if true.

% 36. It is not executable to load and unload the same package with the same airplane at the same location at the same time.
nonexecutable loadAirplane(P,A,L) & unloadAirplane(P,A,L) if true.

% 37. It is not executable for a truck to drive and also be loaded or unloaded at the same time.
nonexecutable drive(T,L1,L2,C) & loadTruck(P,T,L) if true.
nonexecutable drive(T,L1,L2,C) & unloadTruck(P,T,L) if true.

% 38. It is not executable for an airplane to fly and also be loaded or unloaded at the same time.
nonexecutable fly(A,L1,L2) & loadAirplane(P,A,L) if true.
nonexecutable fly(A,L1,L2) & unloadAirplane(P,A,L) if true.


% query
:- query
    0: inCity(l0_0)=c0 & isAirport(l0_0) & inCity(l1_0)=c1 & isAirport(l1_0);
    0: loc(a0)=l1_0 & loc(t0)=l0_0 & loc(t1)=l1_0;
    0: loc(p0)=l1_0 & loc(p1)=l1_0;
    maxstep: loc(p0)=l0_0 & loc(p1)=l0_0.
