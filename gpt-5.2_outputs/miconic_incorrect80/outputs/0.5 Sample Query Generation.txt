% Query 1: Board at the correct origin floor when elevator is there; passenger becomes boarded and not served. (satisfiable)
:- query
    0: isFloor(f0) & isFloor(f1) &
       isPassenger(p0) &
       elevAt(f0) &
       origin(p0)=f0 & dest(p0)=f1 &
       ~boarded(p0) & served(p0);
    0: board(p0,f0);
    maxstep: boarded(p0) & ~served(p0).

% Query 2: Try to board at a floor that is not the passenger's origin; should be disallowed by nonexecutable. (unsatisfiable)
:- query
    0: isFloor(f0) & isFloor(f1) &
       isPassenger(p0) &
       elevAt(f0) &
       origin(p0)=f1 & dest(p0)=f0 &
       ~boarded(p0) & ~served(p0);
    0: board(p0,f0).

% Query 3: Move up from f0 to f1 when above(f0,f1) holds and elevator is at f0; elevator ends at f1 and not at f0. (satisfiable)
:- query
    0: isFloor(f0) & isFloor(f1) &
       above(f0,f1) &
       elevAt(f0);
    0: moveUp(f0,f1);
    maxstep: elevAt(f1) & ~elevAt(f0).

% Query 4: Try to move up when the elevator is not at the source floor; should be disallowed by nonexecutable. (unsatisfiable)
:- query
    0: isFloor(f0) & isFloor(f1) &
       above(f0,f1) &
       elevAt(f1) & ~elevAt(f0);
    0: moveUp(f0,f1).

% Query 5: Board and depart the same passenger in the same step; should be disallowed by concurrency constraint. (unsatisfiable)
:- query
    0: isFloor(f0) &
       isPassenger(p0) &
       elevAt(f0) &
       origin(p0)=f0 & dest(p0)=f0 &
       ~boarded(p0) & ~served(p0);
    0: board(p0,f0) & depart(p0,f0).
