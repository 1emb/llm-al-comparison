:- sorts
    loc >> disk;
    loc >> peg;
    integer.


:- objects
    a, b, c :: peg;
    d1, d2, d3 :: disk;
    1, 2, 3 :: integer.


:- variables
    D, D1, D2, D3, D4 :: disk;
    P, P1, P2, P3, P4 :: peg;
    L, L1, L2, L3, L4 :: loc;
    N, N1, N2, N3, N4 :: integer;
    Bool :: boolean.


:- constants
    loc(disk) :: inertialFluent(loc);
    weight(disk) :: inertialFluent(integer);
    onTop(disk) :: inertialFluent(boolean);
    move(disk, loc) :: exogenousAction.

% Generated constraints
% 1. A disk cannot be moved if there is another disk on top of it.
nonexecutable move(D, L) if onTop(D).

% 2. A disk cannot be moved onto another disk if it is heavier than the disk it is being moved onto.
nonexecutable move(D1, D2) if weight(D1) > weight(D2).

% 3. Moving a disk causes its location to change to the new location.
move(D, L) causes loc(D) = L.

% 4. It is impossible for two disks to be at the same location simultaneously.
impossible loc(D1) = L & loc(D2) = L & D1 \= D2.

% 5. The move action is not executable if the location to be moved to is already occupied by another disk or peg.
nonexecutable move(D, L) if loc(D1) = L & D1 \= D.

% 6. A disk is on top if there is no other disk on it.
onTop(D) if ~existsOtherDiskOn(D).

% 7. A disk is not on top if there is another disk on it.
~onTop(D) if existsOtherDiskOn(D).

% Helper static law to determine if there is another disk on top
existsOtherDiskOn(D) if loc(D1) = D & D1 \= D.


noconcurrency.

:- query
    0: loc(d1) = a & loc(d2) = d1 & loc(d3) = d2 & weight(d1) = 3 & weight(d2) = 2 & weight(d3) = 1;
    maxstep: loc(d1) = c & loc(d2) = d1 & loc(d3) = d2.
